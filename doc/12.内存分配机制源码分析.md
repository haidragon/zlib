# 11. 内存分配机制源码分析

## 一、概述

zlib 库的内存管理是其核心机制之一，直接影响压缩/解压性能和内存使用效率。libz-1.2.8 在内存分配方面进行了大量优化，主要包括：

1. **灵活的分配器接口**: 允许用户自定义内存分配函数
2. **内部优化分配**: 默认使用系统 malloc/free，但进行安全封装
3. **流式状态管理**: 每个压缩/解压流维护独立的状态结构
4. **缓冲区预分配**: 减少动态分配次数，提高性能

---

## 二、内存分配接口设计

### 2.1 公开 API - alloc_func 和 free_func

```c
// zlib.h:75-77
typedef void *(*alloc_func)(void *opaque, unsigned int items, 
                           unsigned int size);
typedef void   (*free_func)(void *opaque, void *address);
```

**设计特点**:
- `items` × `size` 参数模式，类似 `calloc` 但无零初始化
- `opaque` 指针允许传递私有上下文
- 返回 `void*` 支持任意类型数据

### 2.2 z_stream 结构中的分配器

```c
// zlib.h:80-105
typedef struct z_stream_s {
    // ... 输入输出缓冲区指针 ...
    
    alloc_func zalloc;  // 用于分配内部状态的函数
    free_func  zfree;   // 用于释放内部状态的函数
    void      *opaque;  // 传递给 zalloc 和 zfree 的私有数据
    
    // ... 其他字段 ...
} z_stream;
```

**关键设计**:
- 每个流可独立配置分配器
- `opaque` 支持内存池、垃圾回收等高级特性
- 分配器在流初始化时设置，整个生命周期不变

## 三、默认分配器实现 (zutil.c)

### 3.1 默认分配函数

```c
// zutil.c:97-103
ZLIB_INTERNAL void *zcalloc(void *opaque, unsigned int items, 
                           unsigned int size)
{
    (void) opaque;  // 忽略 opaque，直接使用系统分配器
    return calloc((size_t) items, (size_t) size);
}
```

**注意**: 虽然名为 `calloc`，但实际上 libz 不使用零初始化优势，只是兼容接口。

### 3.2 默认释放函数

```c
// zutil.c:106-112
ZLIB_INTERNAL void zcfree(void *opaque, void *ptr)
{
    (void) opaque;
    free(ptr);
}
```

### 3.3 内部安全分配封装

```c
// zutil.c:114-127
ZLIB_INTERNAL void *z_stream_alloc(z_stream *strm, size_t size)
{
    if (size == 0)
        size = 1;  // 避免分配 0 字节
    
    // 检测是否使用默认分配器
    if (strm->zalloc == (alloc_func)0 || strm->zalloc == zcalloc)
        return malloc(size);  // 直接使用 malloc
    
#if UINT_MAX < SIZE_MAX
    // 安全检查：防止乘法溢出
    if (UINT_MAX < size)
        return NULL;
#endif
    
    // 调用用户自定义分配器
    return strm->zalloc(strm->opaque, (unsigned int) size, 1);
}
```

**安全机制**:
1. **零大小处理**: 将 0 字节请求转为 1 字节，避免未定义行为
2. **溢出检测**: 检查 `size` 是否超过 `unsigned int` 范围
3. **快速路径**: 默认分配器直接调用 `malloc`，跳过函数调用开销

### 3.4 数组分配封装

```c
// zutil.c:129-135
ZLIB_INTERNAL void *z_stream_allocarray(z_stream *strm, 
                                        size_t nmemb, size_t size)
{
    // 乘法溢出检查
    if (size && nmemb && SIZE_MAX / size < nmemb)
        return errno = ENOMEM, (void *) NULL;
    
    return z_stream_alloc(strm, nmemb * size);
}
```

**溢出检查逻辑**:
```c
if (SIZE_MAX / size < nmemb) {
    // nmemb * size > SIZE_MAX，会溢出
    return NULL;
}
```

### 3.5 安全释放封装

```c
// zutil.c:137-147
ZLIB_INTERNAL void z_stream_free(z_stream *strm, void *ptr)
{
    // 检测默认释放函数
    if (strm->zfree == (free_func)0 || strm->zfree == zcfree) {
        free(ptr);
        return;
    }
    
    // 空指针检查
    if (!ptr)
        return;
    
    // 调用用户自定义释放函数
    strm->zfree(strm->opaque, ptr);
}
```

**设计考虑**:
- 自动处理 `NULL` 指针，即使用户分配器不检查
- 默认分配器使用系统 `free`，提高效率

### 4.2 deflate 状态初始化

```c
// deflate.c:1580-1700
int ZEXPORT deflateInit2_(z_stream *strm, int level, int method,
                         int windowBits, int memLevel, 
                         int strategy, const char *version, 
                         int stream_size)
{
    deflate_state *s;
    
    // === 步骤 1: 分配 deflate_state 结构 ===
    s = (deflate_state *)z_stream_alloc(strm, 
                sizeof(deflate_state));
    if (s == NULL) {
        return Z_MEM_ERROR;
    }
    
    // 初始化基本字段
    s->strm = strm;
    strm->state = (struct internal_state *)s;
    s->status = INIT_STATE;
    
    // === 步骤 2: 计算窗口大小 ===
    s->w_bits = windowBits;  // 8-15
    s->w_size = 1 << s->w_bits;  // 256B - 32KB
    s->w_mask = s->w_size - 1;
    
    // === 步骤 3: 分配滑动窗口 ===
    s->window = (unsigned char *)z_stream_allocarray(
                    strm, 1, s->w_size * 2);
    // 窗口大小为 2*w_size，用于循环缓冲
    
    // === 步骤 4: 计算哈希表参数 ===
    s->hash_bits = memLevel + 7;  // memLevel 影响哈希表大小
    s->hash_size = 1 << s->hash_bits;
    s->hash_mask = s->hash_size - 1;
    s->hash_shift = ((s->hash_bits + MIN_MATCH - 1) / MIN_MATCH);
    
    // === 步骤 5: 分配哈希表 ===
    s->head = (Pos *)z_stream_allocarray(
                  strm, s->hash_size, sizeof(Pos));
    s->prev = (Pos *)z_stream_allocarray(
                  strm, s->w_size, sizeof(Pos));
    
    // === 步骤 6: 分配挂起缓冲区 ===
    s->pending_buf_size = (unsigned long)memLevel * 1024;
    s->pending_buf = (unsigned char *)z_stream_allocarray(
                         strm, 1, s->pending_buf_size);
    
    // === 步骤 7: 初始化 Huffman 树 ===
    s->dyn_ltree = (ct_data *)z_stream_allocarray(
                       strm, HEAP_SIZE, sizeof(ct_data));
    s->dyn_dtree = (ct_data *)z_stream_allocarray(
                       strm, D_CODES*2+1, sizeof(ct_data));
    s->bl_tree = (ct_data *)z_stream_allocarray(
                     strm, 2*BL_CODES+1, sizeof(ct_data));
    
    // ... 其他初始化 ...
    
    return Z_OK;
}
```

### 4.3 内存分配量计算示例

假设使用默认参数：
- `windowBits = 15` → 窗口 32KB
- `memLevel = 8` → 标准内存级别

**各组件内存占用**:

```c
// deflate_state 结构本身
sizeof(deflate_state) ≈ 400 字节

// 滑动窗口 (2 × 32KB)
2 × 32768 = 65,536 字节 = 64KB

// 哈希表 head[]
hash_size = 2^(8+7) = 2^15 = 32,768 项
32,768 × 2 字节 (Pos=unsigned short) = 65,536 字节 = 64KB

// 前向指针 prev[]
w_size = 32,768 项
32,768 × 2 字节 = 65,536 字节 = 64KB

// 挂起缓冲区
memLevel × 1024 = 8 × 1024 = 8,192 字节 = 8KB

// Huffman 树
dyn_ltree: 2×L_CODES+1 = 2×286+1 = 573 项
dyn_dtree: 2×D_CODES+1 = 2×30+1 = 61 项
bl_tree: 2×BL_CODES+1 = 2×19+1 = 39 项
每项 4 字节 (ct_data)，共约 2.7KB

总计：约 200KB
```

### 4.4 deflate 状态释放

```c
// deflate.c:1730-1750
int ZEXPORT deflateEnd(z_stream *strm)
{
    deflate_state *s;
    
    if (strm == NULL || strm->state == NULL)
        return Z_STREAM_ERROR;
    
    s = (deflate_state *)strm->state;
    
    // === 释放所有分配的内存 ===
    
    // 1. 释放 Huffman 树
    z_stream_free(strm, s->dyn_ltree);
    z_stream_free(strm, s->dyn_dtree);
    z_stream_free(strm, s->bl_tree);
    
    // 2. 释放挂起缓冲区
    z_stream_free(strm, s->pending_buf);
    
    // 3. 释放哈希表和窗口
    z_stream_free(strm, s->head);
    z_stream_free(strm, s->prev);
    z_stream_free(strm, s->window);
    
    // 4. 释放状态结构本身
    z_stream_free(strm, s);
    
    // 清空流状态
    strm->state = NULL;
    
    return Z_OK;
}
```

### 6.2 gz_open 内存分配流程

```c
// gzlib.c:24-150
static gzFile gz_open(const char *path, int fd, const char *mode)
{
    gz_state *state;
    
    // === 步骤 1: 分配 gz_state 结构 ===
    state = (gz_state *)malloc(sizeof(gz_state));
    if (state == NULL)
        return NULL;
    
    state->size = 0;        // 尚未分配缓冲区
    state->want = GZBUFSIZE; // 期望大小 8192
    state->msg = NULL;
    state->path = strdup(path);  // 复制路径字符串
    
    // === 步骤 2: 解析模式字符串 ===
    while (*mode) {
        if (*mode >= '0' && *mode <= '9')
            state->level = *mode - '0';
        else if (*mode == 'r')
            state->mode = GZ_READ;
        else if (*mode == 'w')
            state->mode = GZ_WRITE;
        // ... 其他模式
        mode++;
    }
    
    // === 步骤 3: 打开文件 ===
    state->fd = open(path, oflag, 0666);
    if (state->fd == -1) {
        free(state->path);
        free(state);
        return NULL;
    }
    
    // === 步骤 4: 延迟分配缓冲区 ===
    // 缓冲区在实际读写时才分配 (惰性分配)
    // 见 gz_comp() 和 gz_decompress()
    
    return (gzFile)state;
}
```

### 6.3 缓冲区惰性分配

```c
// gzlib.c:200-250
local int gz_init(gz_statep state)
{
    // === 按需分配输入输出缓冲区 ===
    
    if (state->mode == GZ_READ) {
        // 读模式：分配双缓冲区
        state->size = state->want * 2;
        state->in = (unsigned char *)malloc(state->size);
        state->out = (unsigned char *)malloc(state->size);
    } else {
        // 写模式：单缓冲区
        state->size = state->want;
        state->in = (unsigned char *)malloc(state->size);
        state->out = NULL;  // 写模式不需要输出缓冲
    }
    
    if (state->in == NULL) {
        gz_error(state, Z_MEM_ERROR, "out of memory");
        return -1;
    }
    
    // 初始化内嵌 zlib 流
    state->strm.zalloc = zcalloc;
    state->strm.zfree = zcfree;
    state->strm.opaque = Z_NULL;
    
    return 0;
}
```

**惰性分配的优势**:
- 避免不必要的内存分配
- 根据实际模式 (读/写) 优化缓冲区大小
- 支持动态调整缓冲区大小

## 七、自定义分配器应用

### 7.1 使用内存池

```c
#include <zlib.h>

#define POOL_SIZE (1024 * 1024)  // 1MB 池

typedef struct {
    unsigned char pool[POOL_SIZE];
    size_t offset;
} memory_pool;

void *pool_alloc(void *opaque, unsigned int items, unsigned int size)
{
    memory_pool *pool = (memory_pool *)opaque;
    size_t total = items * size;
    
    // 简单线性分配
    if (pool->offset + total > POOL_SIZE)
        return NULL;  // 池耗尽
    
    void *ptr = pool->pool + pool->offset;
    pool->offset += total;
    return ptr;
}

void pool_free(void *opaque, void *address)
{
    // 内存池通常不单独释放，重置整个池
}

// 使用示例
int compress_with_pool(unsigned char *input, size_t input_len,
                       unsigned char *output, size_t *output_len)
{
    z_stream strm;
    memory_pool pool = {0};  // 初始化池
    
    strm.zalloc = pool_alloc;
    strm.zfree = pool_free;
    strm.opaque = &pool;
    
    int ret = deflateInit(&strm, Z_DEFAULT_COMPRESSION);
    // ... 执行压缩 ...
    deflateEnd(&strm);
    
    // 无需手动释放，池会自动回收
    return ret;
}
```

### 7.2 带统计的分配器

```c
typedef struct {
    size_t total_allocated;
    size_t peak_usage;
    size_t current_usage;
    int alloc_count;
} alloc_stats;

void *stats_alloc(void *opaque, unsigned int items, unsigned int size)
{
    alloc_stats *stats = (alloc_stats *)opaque;
    void *ptr = malloc(items * size);
    
    if (ptr) {
        stats->total_allocated += items * size;
        stats->current_usage += items * size;
        if (stats->current_usage > stats->peak_usage)
            stats->peak_usage = stats->current_usage;
        stats->alloc_count++;
    }
    
    return ptr;
}

void stats_free(void *opaque, void *address)
{
    alloc_stats *stats = (alloc_stats *)opaque;
    // 实际应用中需要记录每个分配的大小
    free(address);
}

// 避免频繁创建销毁流
z_stream strm;
deflateInit(&strm, Z_DEFAULT_COMPRESSION);

// 多次复用
for (int i = 0; i < 100; i++) {
    deflateReset(&strm);  // 重置状态但不释放内存
    // ... 执行压缩 ...
}

deflateEnd(&strm);  // 最后统一释放
```

## 九、内存泄漏检测

### 9.1 常见泄漏点

```c
// 错误示例：忘记调用 deflateEnd
void bad_compress() {
    z_stream strm;
    deflateInit(&strm, Z_DEFAULT_COMPRESSION);
    // ... 压缩操作 ...
    // 缺少：deflateEnd(&strm);  ← 内存泄漏!
}

// 正确做法
void good_compress() {
    z_stream strm;
    int ret = deflateInit(&strm, Z_DEFAULT_COMPRESSION);
    if (ret != Z_OK) return;
    
    // ... 压缩操作 ...
    
    deflateEnd(&strm);  // 确保释放
}
```

### 9.2 异常安全处理

```c
int safe_compress(unsigned char *input, size_t input_len,
                  unsigned char *output, size_t *output_len)
{
    z_stream strm;
    int ret = Z_OK;
    
    strm.zalloc = Z_NULL;
    strm.zfree = Z_NULL;
    strm.opaque = Z_NULL;
    
    ret = deflateInit(&strm, Z_DEFAULT_COMPRESSION);
    if (ret != Z_OK) goto error;
    
    strm.next_in = input;
    strm.avail_in = input_len;
    strm.next_out = output;
    strm.avail_out = *output_len;
    
    ret = deflate(&strm, Z_FINISH);
    if (ret != Z_STREAM_END) {
        ret = (ret == Z_OK) ? Z_BUF_ERROR : ret;
        goto cleanup;
    }
    
    *output_len = strm.total_out;
    
cleanup:
    deflateEnd(&strm);  // 总是执行
error:
    return ret;
}
```

## 十一、总结与最佳实践

### 11.1 内存分配原则

1. **始终配对调用**: `deflateInit` ↔ `deflateEnd`, `inflateInit` ↔ `inflateEnd`
2. **检查返回值**: 所有初始化函数都可能失败
3. **使用默认分配器**: 除非有特殊需求，否则使用系统 malloc/free
4. **复用流对象**: 避免频繁创建销毁

### 11.2 内存配置建议

**嵌入式系统** (内存受限):
```c
deflateInit2(&strm, level, Z_DEFLATED, 
             12,  // 4KB 窗口
             4,   // 较低 memLevel
             Z_DEFAULT_STRATEGY);
```

**服务器应用** (性能优先):
```c
deflateInit2(&strm, level, Z_DEFLATED,
             15,  // 32KB 窗口
             9,   // 最高 memLevel
             Z_DEFAULT_STRATEGY);
```

**通用场景** (平衡):
```c
deflateInit(&strm, Z_DEFAULT_COMPRESSION);
// 等价于:
// deflateInit2(&strm, level, Z_DEFLATED, 15, 8, 0);
```
