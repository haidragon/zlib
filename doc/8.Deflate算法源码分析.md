# Deflate算法源码分析

## 一、Deflate算法概述

Deflate是zlib使用的核心压缩算法，由LZ77字典编码和霍夫曼统计编码组成。

```
Deflate算法组成:
┌─────────────────────────────────────┐
│           Deflate压缩               │
├─────────────────┬───────────────────┤
│   LZ77阶段      │   霍夫曼阶段      │
│  字典编码        │   统计编码         │
├─────────────────┼───────────────────┤
│ • 滑动窗口       │ • 频率统计         │
│ • 字符串匹配     │ • 霍夫曼树构建     │
│ • 距离/长度对    │ • 变长编码         │
└─────────────────┴───────────────────┘
```

```
deflate相关源码:
```

## 二、Deflate初始化

### 2.1 deflateInit2_()函数

```c
// 在deflate.c中
int ZEXPORT deflateInit2_(z_stream *strm,
                         int level,
                         int method,
                         int windowBits,
                         int memLevel,
                         int strategy,
                         const char *version,
                         int stream_size) {
    deflate_state *s;
    int wrap = 1;
    static const char my_version[] = ZLIB_VERSION;

    // 版本检查
    if (version == Z_NULL || version[0] != my_version[0] ||
        stream_size != sizeof(z_stream)) {
        return Z_VERSION_ERROR;
    }

    // 参数检查
    if (strm == Z_NULL) return Z_STREAM_ERROR;

    // 方法检查
    if (method != Z_DEFLATED) return Z_STREAM_ERROR;

    // 窗口位检查
    if (windowBits < 0) {
        wrap = 0;
        windowBits = -windowBits;
    }
    if (memLevel < 1 || memLevel > MAX_MEM_LEVEL ||
        windowBits < 8 || windowBits > 15 ||
        level < 0 || level > 9) {
        return Z_STREAM_ERROR;
    }

    // 分配deflate_state
    s = (deflate_state *)ZALLOC(strm, 1, sizeof(deflate_state));
    if (s == Z_NULL) return Z_MEM_ERROR;
    strm->state = (struct internal_state *)s;
    s->strm = strm;

    // 初始化窗口参数
    s->w_size = 1 << (windowBits - 1);
    s->w_bits = windowBits - 1;
    s->w_mask = s->w_size - 1;

    // 初始化hash参数
    s->hash_size = 1 << s->hash_bits;
    s->hash_mask = s->hash_size - 1;

    // 分配窗口缓冲区
    s->window = (Bytef *)ZALLOC(strm, s->w_size * 2, 2);
    s->pending_buf = (uchf *)ZALLOC(strm, (ulg)s->w_size, (sizeof(int)+2));
    s->prev = (Posf *)ZALLOC(strm, s->w_size, sizeof(Pos));
    s->head = (Posf *)ZALLOC(strm, s->hash_size, sizeof(Pos));

    // 初始化压缩参数
    s->wrap = wrap;
    s->gzhead = Z_NULL;
    s->status = INIT_STATE;
    s->last_flush = Z_NO_FLUSH;

    // 根据压缩级别选择配置
    s->max_lazy_match = configuration_table[level].max_lazy;
    s->good_match = configuration_table[level].good_length;
    s->nice_match = configuration_table[level].nice_length;
    s->max_chain_length = configuration_table[level].max_chain;

    // 初始化匹配状态
    lm_init(s);

    return Z_OK;
}
```

## 三、Deflate主函数

### 3.1 deflate()函数

```c
// 在deflate.c中
int ZEXPORT deflate(z_stream *strm, int flush) {
    int old_flush; /* value of flush param for previous deflate call */
    deflate_state *s;

    // 参数检查
    if (strm == Z_NULL || strm->state == Z_NULL) {
        return Z_STREAM_ERROR;
    }
    s = strm->state;

    // Flush参数检查
    if (flush > Z_BLOCK || flush < 0) {
        return Z_STREAM_ERROR;
    }
    old_flush = s->last_flush;
    s->last_flush = flush;

    // 处理特殊状态
    if (s->status == INIT_STATE) {
        if (s->wrap == 2) {
            // GZIP头部
            put_byte(s, 31);
            put_byte(s, 139);
            put_byte(s, 8);
            if (s->gzhead == Z_NULL) {
                put_byte(s, 0);
                put_byte(s, 0);
                put_byte(s, 0);
                put_byte(s, 0);
                put_byte(s, 0);
                put_byte(s, 0);
                put_byte(s, OS_CODE);
                s->status = BUSY_STATE;
            } else {
                // 有自定义GZIP头部
                s->gzindex = 0;
                s->status = EXTRA_STATE;
            }
        } else if (s->wrap == 1) {
            // ZLIB头部
            unsigned int header = (Z_DEFLATED + ((s->w_bits - 8) << 4)) << 8;
            unsigned int level_flags = (s->level - 1) >> 1;
            if (level_flags > 3) level_flags = 3;
            header |= (level_flags << 6);
            if (s->strstart != 0) header |= PRESET_DICT;
            header += 31 - (header % 31);
            putShortMSB(s, header);
            if (s->strstart != 0) {
                putShortMSB(s, s->adler >> 16);
                putShortMSB(s, s->adler & 0xffff);
            }
            s->status = BUSY_STATE;
        } else {
            s->status = BUSY_STATE;
        }
    }

    // 压缩主循环
    if (s->status == BUSY_STATE) {
        if (s->avail_in != 0 || s->lookahead != 0 ||
            (flush != Z_NO_FLUSH && s->status != FINISH_STATE)) {
            block_state bstate;

            // 根据配置选择压缩函数
            bstate = s->strategy == Z_HUFFMAN_ONLY ? deflate_huff(s, flush) :
                      s->strategy == Z_RLE ? deflate_rle(s, flush) :
                      (*(configuration_table[s->level].func))(s, flush);

            if (bstate == finish_started || bstate == finish_done) {
                s->status = FINISH_STATE;
            }
            if (bstate == need_more || bstate == finish_started) {
                if (strm->avail_out == 0) {
                    return Z_BUF_ERROR;
                }
                break;
            }
            if (bstate == block_done) {
                if (flush == Z_PARTIAL_FLUSH) {
                    tr_align(s);
                } else if (flush != Z_BLOCK) {
                    tr_flush_block(s, 0, 0);
                }
                if (flush == Z_PACKET_FLUSH) {
                    tr_flush_block(s, 0, 0);
                }
                if (flush == Z_SYNC_FLUSH) {
                    tr_flush_bits(s);
                }
                if (flush == Z_FULL_FLUSH) {
                    tr_reset(s);
                }
            }
        }
    }

    // 完成状态
    if (s->status == FINISH_STATE) {
        if (s->wrap > 0) {
            // 写入尾部
            put_byte(s, s->wrap == 2 ? 0 : 0xff);
            if (s->wrap == 2) {
                put_byte(s, 0xff);
                put_byte(s, 0);
                put_byte(s, 0);
                put_byte(s, 0);
            }
            putShortMSB(s, s->adler >> 16);
            putShortMSB(s, s->adler & 0xffff);
        }
        s->status = BUSY_STATE;
        flush_pending(strm);
        if (s->pending != 0) {
            return Z_BUF_ERROR;
        }
    }

    // 更新统计信息
    if (flush != Z_FINISH || ret != Z_STREAM_END) {
        return Z_OK;
    }

    return Z_STREAM_END;
}
```

## 四、快速压缩实现

### 4.1 deflate_fast()函数

```c
// 在deflate.c中
static block_state deflate_fast(deflate_state *s, int flush) {
    IPos hash_head;          /* head of the hash chain */
    int bflush;              /* set if current block must be flushed */

    for (;;) {
        /* Make sure that we always have enough lookahead, except
         * at the end of the input file. We need MAX_MATCH bytes
         * for the next match, plus MIN_MATCH bytes to insert the
         * string following the next match.
         */
        if (s->lookahead < MIN_LOOKAHEAD) {
            fill_window(s);
            if (s->lookahead < MIN_LOOKAHEAD && flush == Z_NO_FLUSH) {
                return need_more;
            }
            if (s->lookahead == 0) break; /* flush the current block */
        }

        /* Insert the string window[strstart .. strstart+2] in the
         * dictionary, and set hash_head to the head of the hash chain:
         */
        hash_head = NIL;
        if (s->lookahead >= MIN_MATCH) {
            INSERT_STRING(s, s->strstart, hash_head);
        }

        /* Find the longest match, discarding those <= prev_length.
         * At this point we have always match_length < MIN_MATCH
         */
        if (hash_head != NIL && s->strstart - hash_head <= MAX_DIST(s)) {
            /* To simplify the code, we prevent matches with the string
             * of window index 0 (in particular we have to avoid a match
             * of the string with itself at the start of the input file).
             */
            s->match_length = longest_match(s, hash_head);
            /* longest_match() sets match_start */
        }
        if (s->match_length >= MIN_MATCH) {
            /* 在快速模式中，找到匹配就立即使用 */
            check_match(s, s->strstart, s->match_start, s->match_length);

            bflush = tr_tally_dist(s, s->strstart - s->match_start,
                                   s->match_length - MIN_MATCH);

            s->lookahead -= s->match_length;

            /* Insert new strings in the hash table only if the match length
             * is not too large. This saves time but degrades compression.
             */
#ifndef FASTEST
            if (s->match_length <= s->max_insert_length &&
                s->lookahead >= MIN_MATCH) {
                s->match_length--; /* string at strstart already in hash table */
                do {
                    s->strstart++;
                    INSERT_STRING(s, s->strstart, hash_head);
                    /* strstart never exceeds WSIZE-MAX_MATCH-1 */
                } while (--s->match_length != 0);
                s->strstart++;
            } else {
#else
            {
#endif
                s->strstart += s->match_length;
                s->match_length = 0;
                s->ins_h = s->window[s->strstart];
                UPDATE_HASH(s, s->ins_h, s->window[s->strstart + 1]);
#if MIN_MATCH != 3
                Call UPDATE_HASH() MIN_MATCH-3 more times
#endif
            }
        } else {
            /* 没有找到匹配，输出字面值 */
            bflush = tr_tally_lit(s, s->window[s->strstart]);
            s->lookahead--;
            s->strstart++;
        }
        if (bflush) FLUSH_BLOCK(s, 0);
    }
    s->insert = s->strstart < MIN_MATCH - 1 ? s->strstart : MIN_MATCH - 1;
    if (flush == Z_FINISH) {
        FLUSH_BLOCK(s, 1);
        return finish_done;
    }
    if (s->last_lit)
        FLUSH_BLOCK(s, 0);
    return block_done;
}
```

## 五、慢速压缩实现

### 5.1 deflate_slow()函数

```c
// 在deflate.c中
static block_state deflate_slow(deflate_state *s, int flush) {
    IPos hash_head;          /* head of hash chain */
    int bflush;              /* set if current block must be flushed */

    /* Process the input block. */
    for (;;) {
        /* Make sure that we always have enough lookahead, except
         * at the end of the input file. We need MAX_MATCH bytes
         * for the next match, plus MIN_MATCH bytes to insert the
         * string following the next match.
         */
        if (s->lookahead < MIN_LOOKAHEAD) {
            fill_window(s);
            if (s->lookahead < MIN_LOOKAHEAD && flush == Z_NO_FLUSH) {
                return need_more;
            }
            if (s->lookahead == 0) break; /* flush the current block */
        }

        /* 插入当前字符串到哈希表 */
        hash_head = NIL;
        if (s->lookahead >= MIN_MATCH) {
            INSERT_STRING(s, s->strstart, hash_head);
        }

        /* 查找匹配 */
        s->prev_length = s->match_length;
        s->prev_match = s->match_start;
        s->match_length = MIN_MATCH - 1;

        if (hash_head != NIL && s->prev_length < s->max_lazy_match &&
            s->strstart - hash_head <= MAX_DIST(s)) {
            /* 找到潜在匹配 */
            s->match_length = longest_match(s, hash_head);
            /* longest_match() sets match_start */

            if (s->match_length <= 5 && (s->strategy == Z_FILTERED)) {
                /* 对于过滤策略，短匹配也被忽略 */
                s->match_length = MIN_MATCH - 1;
            }
        }

        /* 懒惰匹配: 检查是否应该延迟 */
        if (s->prev_length >= MIN_MATCH && s->match_length <= s->prev_length) {
            /* 之前的匹配更好，使用它 */
            uInt max_insert = s->strstart + s->lookahead - MIN_MATCH;

            /* 输出之前的匹配 */
            check_match(s, s->strstart - 1, s->prev_match, s->prev_length);
            bflush = tr_tally_dist(s, s->strstart - 1 - s->prev_match,
                                   s->prev_length - MIN_MATCH);

            /* 更新窗口 */
            s->lookahead -= s->prev_length - 1;
            s->prev_length -= 2;
            do {
                if (++s->strstart <= max_insert) {
                    INSERT_STRING(s, s->strstart, hash_head);
                }
            } while (--s->prev_length != 0);
            s->match_available = 0;
            s->match_length = MIN_MATCH - 1;
            s->strstart++;

            if (bflush) FLUSH_BLOCK(s, 0);

        } else if (s->match_available) {
            /* 输出保存的字面值 */
            bflush = tr_tally_lit(s, s->window[s->strstart - 1]);
            if (bflush) {
                FLUSH_BLOCK(s, 0);
            }
            s->strstart++;
            s->lookahead--;
            if (s->strstart + s->lookahead <= s->w_size + MIN_LOOKAHEAD) {
                s->match_length = MIN_MATCH - 1;
            }
        } else {
            /* 保存当前字面值，继续查找匹配 */
            s->match_available = 1;
            s->strstart++;
            s->lookahead--;
        }

        /* 刷新块 */
        if (s->match_available || s->lookahead == 0 ||
            flush == Z_FINISH || flush == Z_BLOCK) {
            if (s->match_available) {
                bflush = tr_tally_lit(s, s->window[s->strstart - 1]);
                s->match_available = 0;
            }
            if (bflush) FLUSH_BLOCK(s, 0);
        }
    }
    if (flush == Z_FINISH) {
        FLUSH_BLOCK(s, 1);
        return finish_done;
    }
    if (s->last_lit)
        FLUSH_BLOCK(s, 0);
    return block_done;
}
```

### 7.1 块类型

```c
// Deflate数据块类型:
// BFINAL: 最后一个块标志 (1位)
// BTYPE:  块类型 (2位)
//
// BTYPE = 00: 未压缩块
// BTYPE = 01: 固定霍夫曼编码
// BTYPE = 10: 动态霍夫曼编码
// BTYPE = 11: 保留

// 在trees.c中
static void send_bits(deflate_state *s, int value, int length) {
    // 发送指定位数的值
    // bit_buf: 位缓冲区
    // bi_valid: 位缓冲区中的有效位数

    if (s->bi_valid > (int)Buf_size - length) {
        s->bi_buf |= (ush)value << s->bi_valid;
        put_short(s, s->bi_buf);
        s->bi_buf = (ush)value >> (Buf_size - s->bi_valid);
        s->bi_valid += length - Buf_size;
    } else {
        s->bi_buf |= (ush)value << s->bi_valid;
        s->bi_valid += length;
    }
}
```

### 7.2 发送数据块

```c
// 在trees.c中
void ZLIB_INTERNAL tr_flush_block(deflate_state *s, char *buf,
                                   ulg stored_len, int last) {
    ulg opt_lenb, static_lenb; /* opt_len and static_len in bytes */
    int max_blindex = 0;  /* index of last bit length code of non zero freq */

    /* Build the Huffman trees unless a stored block is forced */
    if (s->level > 0) {
        /* 检查是否应该使用存储块 */
        opt_lenb = (ulg)s->opt_len;
        static_lenb = (ulg)s->static_len;

        if (stored_len + 4 <= opt_lenb && stored_len + 4 <= static_lenb) {
            // 使用存储块更小
            send_tree(s, (ct_data *)0, 0);
            compress_block(s, (const ct_data *)0, (const ct_data *)0);
            return;
        }

        /* 发送块类型 */
        send_bits(s, (DYN_TREES << 1) + last, 3);

        /* 发送所有树 */
        send_all_trees(s, s->l_desc.max_code + 1,
                       s->d_desc.max_code + 1,
                       max_blindex + 1);

        /* 压缩数据块 */
        compress_block(s, s->l_desc.dyn_tree, s->d_desc.dyn_tree);

        /* 不包含存储块头部开销 */
        opt_lenb += 3 + 10 + 10 + s->max_lazy_match;

    } else {
        /* level 0: 存储块 */
        send_bits(s, (STORED_BLOCK << 1) + last, 3);
        s->compressed_len += 3;
        opt_lenb = static_lenb = stored_len + 4;
    }

    /* 更新统计 */
    s->compressed_len += opt_lenb;
}
```

## 八、性能优化

### 8.1 匹配查找优化

```c
// 最长匹配查找优化
static unsigned int longest_match(deflate_state *s, IPos cur_match) {
    unsigned chain_length = s->max_chain_length;
    IPos scan = s->strstart;
    IPos match;
    unsigned int len;
    unsigned int best_len = s->prev_length;
    unsigned int nice_match = s->nice_match;
    unsigned int wmask = s->w_mask;
    unsigned char *scan_end1 = s->window + scan + best_len - 1;
    unsigned char *scan_end = s->window + scan + best_len;
    IPos limit;

    /* 计算距离限制 */
    if (s->strstart > MAX_DIST(s)) {
        limit = s->strstart - MAX_DIST(s);
    } else {
        limit = NIL;
    }

    /* 快速检查优化 */
    match = cur_match;
    do {
        /* 首先检查首尾字符 */
        if (match <= limit) break;
        if (s->window[match + best_len] != *scan_end ||
            s->window[match] != s->window[scan]) {
            match = s->prev[match & wmask];
            chain_length--;
            continue;
        }

        /* 完整比较 */
        len = 0;
        while (s->window[match + len] == s->window[scan + len] &&
               len < MAX_MATCH) {
            len++;
        }

        /* 更新最佳匹配 */
        if (len > best_len) {
            s->match_start = match;
            best_len = len;
            if (len >= nice_match) {
                break;
            }
            scan_end1 = s->window + scan + best_len - 1;
            scan_end = s->window + scan + best_len;
        }

        match = s->prev[match & wmask];
        chain_length--;

    } while (chain_length != 0 && match != NIL);

    return best_len;
}
```

### 9.1 检查匹配

```c
#ifdef ZLIB_DEBUG
static void check_match(deflate_state *s, IPos start, IPos match, int length) {
    /* 检查匹配是否正确 */
    if (z_verbose < 2) return;

    fprintf(stderr, "[%ld,%ld]", (long)start, (long)match);

    if (start != match) {
        // 检查字符串是否匹配
        do {
            if (s->window[start++] != s->window[match++]) {
                fprintf(stderr, " mismatch");
                break;
            }
        } while (--length != 0);
    }
}
#endif
```
