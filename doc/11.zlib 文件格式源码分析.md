# 10. zlib 文件格式源码分析

## 一、概述

zlib 库支持三种主要的文件格式：
1. **zlib 格式** (RFC 1950) - 默认的内存压缩格式
2. **gzip 格式** (RFC 1952) - 用于文件压缩的格式
3. **deflate 原始流** (RFC 1951) - 无包装的纯压缩数据流

本文档深入分析这三种格式在源码层面的实现细节。

---

## 二、zlib 格式 (RFC 1950)

### 2.1 格式结构

```
字节序：MSB first (Big Endian)

+-------------------+====================+==================+
| zlib header       | compressed data    | Adler-32 checksum|
| (2 bytes)         | (deflate stream)   | (4 bytes)        |
+-------------------+====================+==================+
```

### 2.2 头部格式详解

#### 第一个字节 (CMF - Compression Method and Flags)

```c
位：7-4 : CM (Compression Method)
     0-8 = DEFLATE (唯一支持的算法)
     
位：3-0 : CINFO (Compression Information)
     0-15: 窗口大小的对数 (log2(window_size))
     CINFO=7 表示 window_size = 32KB
```

#### 第二个字节 (FLG - FLags)

```c
位：7-5 : FCHECK (检查位)
     (CMF*256 + FLG) % 31 == 0
     
位：4   : FDICT (预定义字典标志)
     1 = 使用预定义字典
     0 = 无字典
     
位：3   : FLEVEL (压缩级别)
     00 = 最快压缩
     01 = 快速压缩
     10 = 默认压缩
     11 = 最大压缩
     
位：1-0 : 保留位
```

### 2.3 源码实现位置

#### 写入 zlib 头部 - `deflate.c`

```c
// deflate.c:1593-1622
local int deflateWriteHeader(deflate_state *s, z_stream *strm)
{
    unsigned char header[2];
    
    // CMF 字节：CM=8 (DEFLATE), CINFO=7 (32KB 窗口)
    header[0] = (8 << 4) | 7;
    
    // FLG 字节：计算 FCHECK 使 (header[0]*256 + header[1]) % 31 == 0
    header[1] = 0;
    while ((header[0] * 256 + header[1]) % 31 != 0) {
        header[1]++;
    }
    
    // 设置 FLEVEL 位 (根据压缩级别)
    // 0,1: 最快/快速 -> 00
    // 2-4: 默认 -> 01  
    // 5-7: 更好 -> 10
    // 8-9: 最大 -> 11
    if (s->level >= 8) {
        header[1] |= 0x03;  // 11
    } else if (s->level >= 5) {
        header[1] |= 0x02;  // 10
    } else if (s->level >= 2) {
        header[1] |= 0x01;  // 01
    }
    
    // 如果使用了预定义字典，设置 FDICT 位
    if (s->wrap & 2) {  // gzip 模式
        header[1] |= 0x20;
    }
    
    // 写入头部到输出流
    put_byte(s, header[0]);
    put_byte(s, header[1]);
    
    return Z_OK;
}
```

#### 读取 zlib 头部 - `inflate.c`

```c
// inflate.c:358-410 - inflateResetKeep 中处理头部
local int inflateReadHeader(struct inflate_state *state, 
                           z_stream *strm)
{
    unsigned char header[2];
    
    // 读取 CMF 和 FLG
    header[0] = strm->next_in[0];
    header[1] = strm->next_in[1];
    
    // 验证 CM (必须是 8 = DEFLATE)
    if ((header[0] & 0x0f) != 8) {
        return Z_DATA_ERROR;
    }
    
    // 验证 CINFO (窗口大小，必须<= 7 即 32KB)
    if ((header[0] >> 4) > 7) {
        return Z_DATA_ERROR;
    }
    
    // 验证检查位：(CMF*256 + FLG) % 31 == 0
    if ((((unsigned)header[0] << 8) + header[1]) % 31 != 0) {
        return Z_DATA_ERROR;
    }
    
    // 检查 FDICT 位 (预定义字典)
    if (header[1] & 0x20) {
        state->flags |= 0x04;  // 标记需要字典
        // 后续 4 字节是字典 ID (Adler-32)
    }
    
    // 提取 FLEVEL (仅用于信息，不影响解压)
    state->dmax = 32768;  // 从 CINFO 推导最大距离
    
    return Z_OK;
}
```

## 三、gzip 格式 (RFC 1952)

### 3.1 格式结构

```
+---+---+---+---+---+---+---+---+---+---+=====================+
|ID1|ID2|CM |FLG| XTRA  | MTIME |FLG|HCRC| optional fields     |
+---+---+---+---+---+---+---+---+---+---+=====================+
 
+=======================+========+===========================+
| compressed blocks     | CRC32  | uncompressed size (ISIZE) |
+=======================+========+===========================+
```

### 3.2 头部字段详解 (10-12 字节最小头部)

#### 固定字段 (前 10 字节)

| 偏移 | 字段 | 长度 | 说明 |
|------|------|------|------|
| 0    | ID1  | 1    | 0x1F (标识符) |
| 1    | ID2  | 1    | 0x8B (标识符) |
| 2    | CM   | 1    | 压缩方法 (8=DEFLATE) |
| 3    | FLG  | 1    | 标志位 |
| 4-5  | XTRA | 2    | 可选额外字段长度 |
| 6-9  | MTIME| 4    | 修改时间 (Unix 时间戳) |

#### FLG 标志位详解

```c
位 0 (FTEXT): 文件为 ASCII 文本
位 1 (FHCRC): 头部包含 CRC16
位 2 (FEXTRA): 存在可选额外字段
位 3 (FNAME): 存在原始文件名
位 4 (FCOMMENT): 存在注释
位 5-7: 保留
```

### 3.4 gzip 头部写入 - gzwrite.c

```c
// gzwrite.c: 写入 gzip 头部
local int gz_write_header(gz_statep state)
{
    unsigned char hdr[10];
    time_t mtime;
    
    // ID1, ID2
    hdr[0] = 0x1f;
    hdr[1] = 0x8b;
    
    // CM (压缩方法 = 8, DEFLATE)
    hdr[2] = 8;
    
    // FLG (标志位)
    hdr[3] = 0;
    if (state->level != Z_DEFAULT_COMPRESSION) {
        // 根据压缩级别设置 FLEVEL
        if (state->level <= 2) hdr[3] |= 0x04;  // 最快
        else if (state->level <= 5) hdr[3] |= 0x00;  // 快速
        else if (state->level <= 8) hdr[3] |= 0x02;  // 默认
        else hdr[3] |= 0x06;  // 最大
    }
    
    // MTIME (修改时间)
    mtime = time(NULL);
    hdr[4] = (unsigned char)(mtime);
    hdr[5] = (unsigned char)(mtime >> 8);
    hdr[6] = (unsigned char)(mtime >> 16);
    hdr[7] = (unsigned char)(mtime >> 24);
    
    // XFL (额外标志)
    hdr[8] = 0;
    if (state->level == 1) hdr[8] = 4;  // 最快压缩
    else if (state->level == 9) hdr[8] = 2;  // 最大压缩
    
    // OS (操作系统)
    hdr[9] = 255;  // 未知/跨平台
    
    // 写入头部到文件
    write(state->fd, hdr, 10);
    
    return Z_OK;
}
```

### 3.5 gzip 头部读取 - gzread.c

```c
// gzread.c: 读取并解析 gzip 头部
local int gz_read_header(gz_statep state)
{
    unsigned char buf[10];
    int flags;
    unsigned int len;
    
    // 读取前 10 字节
    read(state->fd, buf, 10);
    
    // 验证魔数 ID1=0x1F, ID2=0x8B
    if (buf[0] != 0x1f || buf[1] != 0x8b) {
        return Z_DATA_ERROR;
    }
    
    // 验证压缩方法 (必须是 8)
    if (buf[2] != 8) {
        return Z_DATA_ERROR;
    }
    
    // 获取标志位
    flags = buf[3];
    
    // 跳过 MTIME (4 字节)、XFL (1 字节)、OS (1 字节)
    // 这些字段在解压时不使用
    
    // 处理可选字段
    if (flags & 4) {  // FEXTRA
        read(fd, buf, 2);  // XLEN
        len = buf[0] + (buf[1] << 8);
        lseek(fd, len, SEEK_CUR);  // 跳过额外字段
    }
    
    if (flags & 8) {  // FNAME
        // 读取原始文件名 (以 NULL 结尾)
        do {
            read(fd, buf, 1);
        } while (buf[0] != 0);
    }
    
    if (flags & 16) {  // FCOMMENT
        // 读取注释 (以 NULL 结尾)
        do {
            read(fd, buf, 1);
        } while (buf[0] != 0);
    }
    
    if (flags & 2) {  // FHCRC
        // 跳过 2 字节头部 CRC
        lseek(fd, 2, SEEK_CUR);
    }
    
    return Z_OK;
}
```

## 四、Deflate 原始流 (RFC 1951)

### 4.1 格式特点

- **无头部**: 直接开始压缩数据
- **无尾部校验**: 不包含 Adler-32 或 CRC32
- **纯比特流**: 按比特对齐而非字节对齐

### 4.2 数据块结构

```c
// deflate.c:57-67
typedef enum {
    STORED_BLOCK = 0,   // 存储块 (不压缩)
    STATIC_TREES = 1,   // 使用静态 Huffman 树
    DYNAMIC_TREES = 2,  // 使用动态 Huffman 树
    RESERVED = 3        // 保留 (错误)
} block_type;
```

每个压缩块的结构：

```
比特 0: BFINAL (最后块标志)
比特 1-2: BTYPE (块类型)
比特 3+: 压缩数据 (根据 BTYPE 解码)
```

### 4.3 源码实现 - 块写入

```c
// deflate.c:200-250
local void send_bits(deflate_state *s, int value, int length)
{
    // 将 value 的 length 个比特写入输出流
    if (s->bi_valid > Buf_size - length) {
        s->bi_buf |= (value << s->bi_valid);
        put_short(s, s->bi_buf);
        s->bi_buf = (unsigned)value >> (Buf_size - s->bi_valid);
        s->bi_valid += length - Buf_size;
    } else {
        s->bi_buf |= (value << s->bi_valid);
        s->bi_valid += length;
    }
}

// deflate.c:900-950
local void _tr_stored_block(deflate_state *s, char *buf, 
                           unsigned long stored_len, int last)
{
    // 发送块头
    send_bits(s, (last << 15) | (STORED_BLOCK << 13), 3+2);
    
    // 对齐到字节边界
    bi_windup(s);
    
    // 存储长度和反码长度
    put_short(s, (unsigned short)stored_len);
    put_short(s, (unsigned short)~stored_len);
    
    // 复制原始数据
    memcpy(s->pending_buf + s->pending, buf, stored_len);
    s->pending += stored_len;
}
```

## 七、校验和实现

### 7.1 Adler-32 (zlib 格式)

```c
// adler32.c:100-150
#define BASE 65521L  // 最大小于 2^32 的素数
#define NMAX 5552    // BASE 的最大重复次数

unsigned long ZEXPORT adler32(unsigned long adler, 
                              const unsigned char *buf,
                              unsigned int len)
{
    unsigned long sum2;
    unsigned n;
    
    // 分割 Adler-32: 高 16 位=sum2, 低 16 位=sum1
    sum2 = (adler >> 16) & 0xffff;
    adler &= 0xffff;
    
    // 每 NMAX 个字节减少一次模运算
    while (len > 0) {
        n = len > NMAX ? NMAX : len;
        len -= n;
        
        // 优化循环：累加 sum1 和 sum2
        while (n >= 16) {
            adler += *buf++; sum2 += adler;
            adler += *buf++; sum2 += adler;
            // ... 展开 16 次
            n -= 16;
        }
        
        // 模运算防止溢出
        adler %= BASE;
        sum2 %= BASE;
    }
    
    // 合并结果：sum2 << 16 | sum1
    return (sum2 << 16) | adler;
}
```

### 7.2 CRC32 (gzip 格式)

```c
// crc32.c:200-300
// 使用预计算的 CRC 表加速计算
local unsigned long crc32_little(unsigned long crc,
                                 const unsigned char *buf,
                                 unsigned int len)
{
    crc = crc ^ 0xffffffff;
    
    // 使用 8K 表 (256 项 x 4 字节)
    #define DO1(crc) crc = crc_table[0][(crc ^ *buf++) & 0xff] ^ (crc >> 8)
    #define DO2(crc) DO1(crc); DO1(crc)
    #define DO4(crc) DO2(crc); DO2(crc)
    #define DO8(crc) DO4(crc); DO4(crc)
    
    while (len >= 8) {
        DO8(crc);
        len -= 8;
    }
    
    while (len--) {
        DO1(crc);
    }
    
    return crc ^ 0xffffffff;
}
```

## 十、实践建议

### 10.1 选择合适的格式

- **网络传输**: 使用 zlib 格式 (头部小，有校验)
- **文件存储**: 使用 gzip 格式 (兼容性好，含元数据)
- **性能敏感**: 使用原始 deflate (无头部开销)
- **兼容性要求高**: 使用 gzip (系统普遍支持)

### 10.2 错误处理要点

```c
// 始终检查返回值
ret = deflate(&strm, Z_FINISH);
if (ret != Z_STREAM_END) {
    // 可能的错误:
    // - Z_BUF_ERROR: 输出缓冲区不足
    // - Z_STREAM_ERROR: 流状态错误
    // - Z_DATA_ERROR: 输入数据损坏
}

// 清理资源
deflateEnd(&strm);
```

// 打印头部字节
void print_zlib_header(unsigned char *buf) {
    printf("CMF: 0x%02X (CM=%d, CINFO=%d)\n", 
           buf[0], buf[0] >> 4, buf[0] & 0x0f);
    printf("FLG: 0x%02X (FLEVEL=%d, FDICT=%d)\n",
           buf[1], (buf[1] >> 6) & 0x03, (buf[1] >> 5) & 0x01);
    printf("Check: %s\n",
           (((buf[0] << 8) + buf[1]) % 31 == 0) ? "OK" : "FAIL");
}
