# Huffman编码实现分析

## 一、Huffman编码概述

Huffman编码是一种最优前缀编码方法，通过构建二叉树为出现频率高的符号分配短编码，为低频率符号分配长编码。

```
Huffman编码流程:
┌─────────┐  ┌─────────┐  ┌─────────┐  ┌─────────┐
│ 统计频率 │→│ 构建树   │→│ 生成编码 │→│ 输出数据 │
└─────────┘  └─────────┘  └─────────┘  └─────────┘
```

### 1.1 在zlib中的应用

zlib中的Huffman编码用于DEFLATE算法的第二阶段：

```
DEFLATE = LZ77 + Huffman编码
               ↓
        3棵Huffman树:
        • 字面值/长度树 (286个符号)
        • 距离树 (30个符号)
        • 位长度树 (19个符号)
```

## 四、构建Huffman树

### 4.1 扫描树 - 统计频率

```c
// 在trees.c中
static void scan_tree(deflate_state *s, ct_data *tree, int max_code) {
    int n;                     /* iterates over all tree elements */
    int prevlen = -1;          /* last emitted length */
    int curlen;                /* length of current code */
    int nextlen = tree[0].Len; /* length of next code */
    int count = 0;             /* repeat count of the current length */
    int max_count = 7;         /* max repeat count */
    int min_count = 4;         /* min repeat count */

    /* 遍历所有符号，统计频率 */
    for (n = 0; n <= max_code; n++) {
        curlen = nextlen;
        nextlen = tree[n + 1].Len;

        if (++count < max_count && curlen == nextlen) {
            continue;

        } else if (count < min_count) {
            s->bl_tree[curlen].Freq += count;

        } else if (curlen != 0) {
            if (curlen != prevlen) {
                s->bl_tree[curlen].Freq++;
            }
            s->bl_tree[REP_3_6].Freq++;

        } else if (count <= 10) {
            s->bl_tree[REPZ_3_10].Freq++;

        } else {
            s->bl_tree[REPZ_11_138].Freq++;
        }

        count = 0;
        prevlen = curlen;
        if (nextlen == 0) {
            max_count = 138;
            min_count = 3;
        } else if (curlen == nextlen) {
            max_count = 6;
            min_count = 3;
        } else {
            max_count = 7;
            min_count = 4;
        }
    }
}
```

### 4.2 构建树 - build_tree()

```c
// 在trees.c中
static void build_tree(deflate_state *s, tree_desc *desc) {
    ct_data *tree = desc->dyn_tree;
    const ct_data *stree = desc->stat_desc->static_tree;
    int elems = desc->stat_desc->elems;
    int n, m;          /* iterate over tree elements */
    int max_code = -1; /* largest code with non zero frequency */
    int node;          /* new node being created */

    /* 构建堆 */
    for (n = 0; n < elems; n++) {
        if (tree[n].Freq != 0) {
            s->heap[++s->heap_len] = max_code = n;
            s->depth[n] = 0;
        } else {
            tree[n].Len = 0;
        }
    }

    /* 堆排序: 按频率排序 */
    while (s->heap_len < 2) {
        node = ++max_code;
        tree[node].Freq = 1;
        s->heap[++s->heap_len] = max_code;
        s->depth[node] = 0;
    }

    /* 为堆中的节点设置父节点 */
    s->heap_max = max_code;
    for (n = (s->heap_len >> 1); n >= 1; n--) {
        pqdownheap(s, tree, n);
    }

    /* 构建Huffman树 */
    node = elems;
    do {
        /* 取出两个频率最小的节点 */
        n = s->heap[SMALLEST];
        s->heap[SMALLEST] = s->heap[s->heap_len--];
        pqdownheap(s, tree, SMALLEST);

        m = s->heap[SMALLEST];
        s->heap[SMALLEST] = s->heap[s->heap_len--];
        pqdownheap(s, tree, SMALLEST);

        /* 创建父节点 */
        tree[node].Freq = tree[n].Freq + tree[m].Freq;
        s->depth[node] = (s->depth[n] > s->depth[m] ?
                          s->depth[n] : s->depth[m]) + 1;
        tree[n].Dad = tree[m].Dad = (unsigned short)node;

        /* 将父节点放回堆中 */
        s->heap[++s->heap_len] = node;
        pqdownheap(s, tree, SMALLEST);

        node++;

    } while (s->heap_len >= 2);

    /* 设置最后一个节点的父节点为0 */
    s->heap[--s->heap_max] = s->heap[SMALLEST];

    /* 生成位长度 */
    gen_bitlen(s, desc);

    /* 生成代码 */
    gen_codes(tree, max_code, s->bl_count);
}
```

### 4.3 堆调整 - pqdownheap()

```c
// 在trees.c中
static void pqdownheap(deflate_state *s, ct_data *tree, int k) {
    int v = s->heap[k];
    int j = k << 1;  /* left child of k */

    while (j <= s->heap_len) {
        /* 设置j为较小的子节点 */
        if (j < s->heap_len &&
            smaller(tree, s->heap[j + 1], s->heap[j], s->depth)) {
            j++;
        }

        /* 父节点小于等于子节点，堆性质满足 */
        if (smaller(tree, v, s->heap[j], s->depth)) {
            break;
        }

        /* 交换父节点和子节点 */
        s->heap[k] = s->heap[j];
        k = j;
        j <<= 1;
    }

    s->heap[k] = v;
}

/* 比较两个节点的频率 */
#define smaller(tree, n, m, depth) \
    (tree[n].Freq < tree[m].Freq || \
    (tree[n].Freq == tree[m].Freq && depth[n] <= depth[m]))
```

### 4.4 生成位长度 - gen_bitlen()

```c
// 在trees.c中
static void gen_bitlen(deflate_state *s, tree_desc *desc) {
    ct_data *tree = desc->dyn_tree;
    int max_code = desc->max_code;
    const ct_data *stree = desc->stat_desc->static_tree;
    int extra = desc->stat_desc->extra_bits;
    int base = desc->stat_desc->extra_base;
    int max_length = desc->stat_desc->max_length;
    int h;              /* heap index */
    int n, m;           /* iterate over the tree elements */
    int bits;           /* bit length */
    int xbits;          /* extra bits */
    ush f;              /* frequency */
    int overflow = 0;   /* number of elements with bit length too large */

    for (bits = 0; bits <= MAX_BITS; bits++) {
        s->bl_count[bits] = 0;
    }

    /* 从叶子节点向上计算位长度 */
    tree[s->heap[s->heap_max]].Len = 0;
    for (h = s->heap_max + 1; h < HEAP_SIZE; h++) {
        n = s->heap[h];
        bits = tree[tree[n].Dad].Len + 1;

        if (bits > max_length) {
            bits = max_length;
            overflow++;
        }
        tree[n].Len = (ush)bits;
        s->bl_count[bits]++;

        /* 如果有额外位数，添加到位长度 */
        if (n > max_code) {
            continue;
        }
        if (stree[n].Len != 0) {
            xbits = stree[n].Len;
            f = tree[n].Freq;
        }
    }

    /* 检查溢出并调整 */
    if (overflow == 0) return;

    /* ... 溢出处理代码 ... */
}
```

### 4.5 生成编码 - gen_codes()

```c
// 在trees.c中
static void gen_codes(ct_data *tree, int max_code, unsigned short *bl_count) {
    unsigned short next_code[MAX_BITS + 1]; /* next code value for each bit length */
    unsigned short code = 0;                /* running code value */
    int bits;                              /* bit index */
    int n;                                 /* code index */

    /* 初始化next_code */
    for (bits = 1; bits <= MAX_BITS; bits++) {
        code = (code + bl_count[bits - 1]) << 1;
        next_code[bits] = code;
    }

    /* 为每个符号生成编码 */
    for (n = 0; n <= max_code; n++) {
        int len = tree[n].Len;

        if (len == 0) {
            continue;
        }

        /* 从next_code获取编码，并递增 */
        tree[n].Code = bi_reverse(next_code[len]++, len);
    }
}
```

### 4.6 位反转 - bi_reverse()

```c
// 在trees.c中
static unsigned int bi_reverse(unsigned int code, int len) {
    unsigned int res = 0;

    do {
        res |= code & 1;
        res <<= 1;
        code >>= 1;
        len--;
    } while (len > 0);

    return res >> 1;
}

/* 示例:
   原始: 1101 (长度4)
   反转: 1011

   原因: zdeflate先处理低位，但输出需要先输出高位
*/
```

## 五、发送Huffman树

### 5.1 send_tree()

```c
// 在trees.c中
static void send_tree(deflate_state *s, ct_data *tree, int max_code) {
    int n;                     /* iterates over all tree elements */
    int prevlen = -1;          /* last emitted length */
    int curlen;                /* length of current code */
    int nextlen = tree[0].Len; /* length of next code */
    int count = 0;             /* repeat count of the current length */
    int max_count = 7;         /* max repeat count */
    int min_count = 4;         /* min repeat count */

    /* 树信息头 */
    if (max_code + 1 >= BL_CODES) {
        send_bits(s, REPZ_11_138, 7);
        send_bits(s, max_code + 1 - BL_CODES, 11);
        send_bits(s, 0, 3);
    }

    /* 发送位长度序列 */
    for (n = 0; n <= max_code; n++) {
        curlen = nextlen;
        nextlen = tree[n + 1].Len;

        if (++count < max_count && curlen == nextlen) {
            continue;

        } else if (count < min_count) {
            do {
                send_code(s, curlen, bl_tree);
            } while (--count != 0);

        } else if (curlen != 0) {
            if (curlen != prevlen) {
                send_code(s, curlen, bl_tree);
                count--;
            }
            send_code(s, REP_3_6, bl_tree);
            send_bits(s, count - 3, 2);

        } else if (count <= 10) {
            send_code(s, REPZ_3_10, bl_tree);
            send_bits(s, count - 3, 3);

        } else {
            send_code(s, REPZ_11_138, bl_tree);
            send_bits(s, count - 11, 7);
        }

        count = 0;
        prevlen = curlen;
        if (nextlen == 0) {
            max_count = 138;
            min_count = 3;
        } else if (curlen == nextlen) {
            max_count = 6;
            min_count = 3;
        } else {
            max_count = 7;
            min_count = 4;
        }
    }
}
```

### 5.2 build_bl_tree()

```c
// 在trees.c中
static int build_bl_tree(deflate_state *s) {
    int max_blindex;  /* index of last bit length code of non zero freq */

    /* 扫描字面值/长度树 */
    scan_tree(s, s->l_desc.dyn_tree, s->l_desc.max_code);

    /* 扫描距离树 */
    scan_tree(s, s->d_desc.dyn_tree, s->d_desc.max_code);

    /* 构建位长度树 */
    build_tree(s, (tree_desc *)(&(s->bl_desc)));

    /* 找到最后一个有非零频率的位长度代码 */
    for (max_blindex = BL_CODES - 1; max_blindex >= 3; max_blindex--) {
        if (s->bl_tree[max_blindex].Len != 0) {
            break;
        }
    }

    /* 确保至少有4个代码 */
    return max_blindex + 1;
}
```

### 5.3 send_all_trees()

```c
// 在trees.c中
static void send_all_trees(deflate_state *s, int lcodes, int dcodes,
                           int blcodes) {
    int rank;                     /* index in bl_order */

    /* 发送HLEN, HDIST, HCLEN */
    send_bits(s, lcodes - 257, 5);
    send_bits(s, dcodes - 1, 5);
    send_bits(s, blcodes - 4, 4);

    for (rank = 0; rank < blcodes; rank++) {
        send_bits(s, s->bl_tree[bl_order[rank]].Len, 3);

    /* 发送字面值/长度树 */
    send_tree(s, s->l_desc.dyn_tree, lcodes - 1);

    /* 发送距离树 */
    send_tree(s, s->d_desc.dyn_tree, dcodes - 1);
}
```

## 六、压缩数据块

### 6.1 compress_block()

```c
// 在trees.c中
static void compress_block(deflate_state *s, const ct_data *ltree,
                           const ct_data *dtree) {
    unsigned dist;      /* distance of matched string */
    int lc;             /* match length or unmatched char */
    unsigned lx = 0;    /* running index in l_buf */
    unsigned code;      /* the code to send */
    int extra;          /* extra bits to send */

    if (s->last_lit != 0) {
        do {
            dist = s->d_buf[lx];
            lc = s->l_buf[lx++];

            if (dist == 0) {
                /* 发送字面值 */
                send_code(s, lc, ltree);
            } else {
                /* 发送长度/距离对 */
                code = length_code[lc];
                send_code(s, code + LITERALS + 1, ltree);
                extra = extra_lbits[code];
                if (extra != 0) {
                    lc -= base_length[code];
                    send_bits(s, lc, extra);
                }

                dist--; /* dist is now the match distance - 1 */
                code = d_code(dist);
                send_code(s, code, dtree);
                extra = extra_dbits[code];
                if (extra != 0) {
                    dist -= base_dist[code];
                    send_bits(s, dist, extra);
                }
            }
        } while (lx < s->last_lit);
    }

    /* 发送块结束标记 */
    send_code(s, END_BLOCK, ltree);
}
```

## 七、数据类型检测

### 7.1 detect_data_type()

```c
// 在trees.c中
static int detect_data_type(deflate_state *s) {
    /* black_mask is the bit mask of black-listed bytes
     * set bits 0..6, 14..25, and 28..31
     * 0xf3ffc07f = binary 11110011111111111100000001111111
     */
    unsigned black_mask = 0xf3ffc07fUL;
    int n;

    /* Check for non-textual ("black-listed") bytes. */
    for (n = 0; n <= 31; n++, black_mask >>= 1)
        if ((black_mask & 1) && (s->bl_tree[n].Freq != 0))
            return Z_BINARY;

    /* Check for textual ("white-listed") bytes. */
    for (n = 32; n < LITERALS; n++)
        if (s->bl_tree[n].Freq != 0)
            return Z_TEXT;

    /* There are no "black-listed" or "white-listed" bytes:
     * this stream either is empty or has tolerated ("gray-listed") bytes only.
     */
    return Z_BINARY;
}
```

## 九、完整流程示例

### 9.1 压缩流程

```c
/* 完整的Huffman编码流程:
 *
 * 1. 统计频率
 *    - 遍历所有符号，统计出现次数
 *    - s->l_desc.dyn_tree[n].Freq = count
 *
 * 2. 构建Huffman树
 *    - build_tree(s, &s->l_desc)
 *    - 使用堆算法构建最优树
 *
 * 3. 生成位长度
 *    - gen_bitlen(s, &s->l_desc)
 *    - 为每个符号确定编码长度
 *
 * 4. 生成编码
 *    - gen_codes(s->l_desc.dyn_tree, max_code, bl_count)
 *    - 根据位长度生成实际编码
 *
 * 5. 构建位长度树
 *    - build_bl_tree(s)
 *    - 为位长度序列构建Huffman树
 *
 * 6. 发送所有树
```

## 十、总结

Huffman编码在zlib中的实现关键点：

1. **3棵树**: 字面值/长度树、距离树、位长度树
2. **堆优化**: 使用堆算法构建最优Huffman树
3. **编码压缩**: 使用重复编码优化位长度传输
4. **静态树**: 预定义固定树，避免传输开销

性能优化：

1. **快速查找**: 使用数组索引代替搜索
2. **位操作**: 高效的位操作实现
3. **内存布局**: 紧凑的数据结构
4. **局部变量**: 减少内存访问

实际应用：

- 数据压缩率提升2-5倍
- 编码速度约50-100 MB/s
- 解码速度约100-200 MB/s
- 内存占用约10-20KB
