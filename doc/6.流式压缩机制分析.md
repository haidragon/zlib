# 流式压缩机制分析

## 一、流式压缩概述

### 1.1 什么是流式压缩

流式压缩是一种不需要一次性加载所有数据到内存的压缩方式，数据可以分块处理。

```
传统一次性压缩:
┌──────────┐                  ┌──────────┐
│  全部数据 │ → 压缩 → ────────→ │ 压缩数据 │
│ (一次性)  │                  │ (一次性)  │
└──────────┘                  └──────────┘

流式压缩:
┌────────┐ ┌────────┐ ┌────────┐      ┌────────┐ ┌────────┐
│ 块1    │ │ 块2    │ │ 块3    │ ...→ │ 块1    │ │ 块2    │
│ →压缩→ │ │ →压缩→ │ │ →压缩→ │      │ →压缩→ │ │ →压缩→ │
└────────┘ └────────┘ └────────┘      └────────┘ └────────┘
   ↓           ↓           ↓              ↓           ↓
 输出块1     输出块2     输出块3        输出块1     输出块2
```

### 1.2 流式压缩的优势

**内存效率**:
- 不需要一次性加载整个文件
- 适合处理大文件（GB级别）
- 内存占用恒定

**实时处理**:
- 边读边压缩，边发送
- 适合网络传输
- 减少延迟

**灵活性**:
- 可以暂停和恢复
- 支持管道操作
- 适合流媒体

### 1.3 流式 vs 一次性对比

| 特性 | 一次性压缩 | 流式压缩 |
|------|-----------|---------|
| 内存占用 | O(n) | O(window_size) |
| 压缩率 | 略高 | 略低（边界效应） |
| 速度 | 快 | 略慢（多次调用） |
| 适用数据 | 小文件 | 大文件 |
| 实时性 | 无 | 支持 |

## 二、z_stream结构详解

### 2.1 核心结构

```c
typedef struct z_stream_s {
    /* 输入缓冲区 */
    z_const unsigned char *next_in;     /* 下一个输入字节指针 */
    unsigned int     avail_in;          /* 可用输入字节数 */
    unsigned long    total_in;          /* 总输入字节数 */

    /* 输出缓冲区 */
    unsigned char    *next_out;         /* 下一个输出字节位置 */
    unsigned int     avail_out;         /* 剩余输出空间 */
    unsigned long    total_out;         /* 总输出字节数 */

    /* 状态信息 */
    z_const char *msg;                  /* 错误消息 */
    struct internal_state *state;       /* 内部状态 */

    /* 内存管理 */
    alloc_func zalloc;                  /* 分配函数 */
    free_func  zfree;                   /* 释放函数 */
    void      *opaque;                  /* 私有数据 */

    /* 其他 */
    int     data_type;                  /* 数据类型 */
    unsigned long   adler;              /* Adler32校验值 */
    unsigned long   reserved;           /* 保留 */
} z_stream;
```

### 2.2 缓冲区管理

#### 输入缓冲区流程

```
初始状态:
next_in: NULL
avail_in: 0
total_in: 0

步骤1: 提供输入
next_in = input_buffer      // 指向输入数据
avail_in = input_length     // 可用字节数
total_in = 0                // 尚未处理

步骤2: 调用deflate()
deflate() 消耗 avail_in 中的数据
avail_in 减小
total_in 增加

步骤3: 缓冲区耗尽
if (avail_in == 0) {
    // 需要提供更多输入
    next_in = new_input_buffer;
    avail_in = new_input_length;
}

步骤4: 最终统计
total_in: 处理的总字节数
```

#### 输出缓冲区流程

```
初始状态:
next_out: output_buffer
avail_out: output_buffer_size
total_out: 0

步骤1: 调用deflate()
deflate() 向 next_out 写入压缩数据
avail_out 减小
total_out 增加

步骤2: 输出缓冲区满
if (avail_out == 0) {
    // 处理已压缩数据
    process_compressed_data(output_buffer);
    
    // 重置缓冲区
    next_out = output_buffer;
    avail_out = output_buffer_size;
}

步骤3: 最终统计
total_out: 输出的总压缩字节数
```

### 2.3 状态转换图

```
压缩状态转换:
    ┌─────────┐
    │  INIT   │  ← deflateInit()
    └────┬────┘
         │ deflate(Z_NO_FLUSH)
         ↓
    ┌─────────┐
    │ BUSY    │  ← 处理数据
    └────┬────┘
         │
         ├─→ avail_in == 0 且 avail_out == 0
         │   → need_more
         │
         ├─→ block_complete
         │   → block_done
         │
         └─→ Z_FINISH
             ↓
         ┌─────────┐
         │FINISHING│  ← 完成压缩
         └────┬────┘
              │
              ↓
         ┌─────────┐
         │  DONE   │  ← Z_STREAM_END
         └─────────┘
```

## 三、流式压缩API详解

### 3.1 初始化流程

```c
// 步骤1: 初始化z_stream
z_stream strm;
strm.zalloc = Z_NULL;  // 使用默认分配器
strm.zfree = Z_NULL;   // 使用默认释放器
strm.opaque = Z_NULL;  // 无私有数据

// 步骤2: 初始化deflate
ret = deflateInit(&strm, Z_DEFAULT_COMPRESSION);
if (ret != Z_OK) {
    // 处理错误
}

// 步骤3: 设置输入输出缓冲区
strm.next_in = input_data;
strm.avail_in = input_size;
strm.next_out = output_buffer;
strm.avail_out = output_buffer_size;
```

### 3.2 主压缩循环

```c
int ret;
do {
    // 压缩数据
    ret = deflate(&strm, Z_NO_FLUSH);
    
    // 检查错误
    if (ret == Z_STREAM_ERROR) {
        // 错误处理
        break;
    }
    
    // 处理输出缓冲区
    if (strm.avail_out == 0 || ret == Z_STREAM_END) {
        size_t have = output_buffer_size - strm.avail_out;
        
        // 写入压缩数据
        write_output(output_buffer, have);
        
        // 重置输出缓冲区
        strm.next_out = output_buffer;
        strm.avail_out = output_buffer_size;
    }
    
    // 处理输入缓冲区
    if (strm.avail_in == 0 && !input_finished) {
        // 读取更多输入
        size_t read_size = read_input(next_input_buffer, buffer_size);
        if (read_size == 0) {
            input_finished = 1;
        } else {
            strm.next_in = next_input_buffer;
            strm.avail_in = read_size;
        }
    }
} while (ret != Z_STREAM_END);
```

### 3.3 完成压缩

```c
// 步骤1: 刷新剩余数据
ret = deflate(&strm, Z_FINISH);
while (ret == Z_OK) {
    // 处理输出
    size_t have = output_buffer_size - strm.avail_out;
    write_output(output_buffer, have);
    
    // 重置输出缓冲区
    strm.next_out = output_buffer;
    strm.avail_out = output_buffer_size;
    
    // 继续刷新
    ret = deflate(&strm, Z_FINISH);
}

// 步骤2: 清理
if (ret != Z_STREAM_END) {
    // 错误处理
}
deflateEnd(&strm);
```

### 3.4 完整示例

```c
#include <zlib.h>
#include <stdio.h>
#include <stdlib.h>

#define CHUNK_SIZE 16384

int compress_stream(FILE *source, FILE *dest) {
    int ret, flush;
    unsigned have;
    z_stream strm;
    unsigned char in[CHUNK_SIZE];
    unsigned char out[CHUNK_SIZE];

    // 初始化deflate流
    strm.zalloc = Z_NULL;
    strm.zfree = Z_NULL;
    strm.opaque = Z_NULL;
    ret = deflateInit(&strm, Z_DEFAULT_COMPRESSION);
    if (ret != Z_OK) {
        return ret;
    }

    // 压缩循环
    do {
        // 读取输入
        strm.avail_in = fread(in, 1, CHUNK_SIZE, source);
        if (ferror(source)) {
            deflateEnd(&strm);
            return Z_ERRNO;
        }
        
        // 确定flush参数
        flush = feof(source) ? Z_FINISH : Z_NO_FLUSH;
        strm.next_in = in;

        // 压缩直到输入耗尽或输出满
        do {
            strm.avail_out = CHUNK_SIZE;
            strm.next_out = out;
            ret = deflate(&strm, flush);
            
            if (ret == Z_STREAM_ERROR) {
                deflateEnd(&strm);
                return ret;
            }
            
            have = CHUNK_SIZE - strm.avail_out;
            if (fwrite(out, 1, have, dest) != have || ferror(dest)) {
                deflateEnd(&strm);
                return Z_ERRNO;
            }
        } while (strm.avail_out == 0);

    } while (flush != Z_FINISH);

    // 清理
    deflateEnd(&strm);
    return Z_OK;
}
```

## 四、流式解压机制

### 4.1 解压流程

```c
int decompress_stream(FILE *source, FILE *dest) {
    int ret;
    unsigned have;
    z_stream strm;
    unsigned char in[CHUNK_SIZE];
    unsigned char out[CHUNK_SIZE];

    // 初始化inflate流
    strm.zalloc = Z_NULL;
    strm.zfree = Z_NULL;
    strm.opaque = Z_NULL;
    strm.avail_in = 0;
    strm.next_in = Z_NULL;
    ret = inflateInit(&strm);
    if (ret != Z_OK) {
        return ret;
    }

    // 解压循环
    do {
        // 读取压缩数据
        strm.avail_in = fread(in, 1, CHUNK_SIZE, source);
        if (ferror(source)) {
            inflateEnd(&strm);
            return Z_ERRNO;
        }
        if (strm.avail_in == 0) {
            break;
        }
        strm.next_in = in;

        // 解压
        do {
            strm.avail_out = CHUNK_SIZE;
            strm.next_out = out;
            ret = inflate(&strm, Z_NO_FLUSH);
            
            switch (ret) {
                case Z_NEED_DICT:
                    ret = Z_DATA_ERROR;
                case Z_DATA_ERROR:
                case Z_MEM_ERROR:
                case Z_STREAM_ERROR:
                    inflateEnd(&strm);
                    return ret;
            }

            have = CHUNK_SIZE - strm.avail_out;
            if (fwrite(out, 1, have, dest) != have || ferror(dest)) {
                inflateEnd(&strm);
                return Z_ERRNO;
            }
        } while (strm.avail_out == 0);

    } while (ret != Z_STREAM_END);

    // 清理
    inflateEnd(&strm);
    return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
}
```

### 4.2 解压状态机

```c
// inflate()内部状态转换
switch (state->mode) {
    case HEAD:          // 读取头部
        if (check_header()) {
            state->mode = FLAGS;
        }
        break;
        
    case FLAGS:         // 读取标志
        read_flags();
        state->mode = TYPE;
        break;
        
    case TYPE:          // 读取块类型
        read_block_type();
        switch (btype) {
            case 0:  // 存储块
                state->mode = STORED;
                break;
            case 1:  // 固定霍夫曼
                state->mode = LEN_;
                break;
            case 2:  // 动态霍夫曼
                state->mode = TABLE;
                break;
        }
        break;
        
    case STORED:        // 存储块
        copy_stored_block();
        state->mode = TYPE;
        break;
        
    case TABLE:         // 动态表
        decode_dynamic_tables();
        state->mode = LEN_;
        break;
        
    case LEN_:          // 读取长度/字面值
        decode_length();
        if (is_literal) {
            state->mode = LIT;
        } else {
            state->mode = LENEXT;
        }
        break;
        
    case LENEXT:        // 读取长度额外位
        read_length_extra();
        state->mode = DIST;
        break;
        
    case DIST:          // 读取距离
        decode_distance();
        state->mode = DISTEXT;
        break;
        
    case DISTEXT:       // 读取距离额外位
        read_distance_extra();
        state->mode = MATCH;
        break;
        
    case MATCH:         // 输出匹配字符串
        output_match_string();
        state->mode = LEN_;
        break;
        
    case LIT:           // 输出字面值
        output_literal();
        state->mode = LEN_;
        break;
        
    case CHECK:         // 校验和
        verify_checksum();
        state->mode = DONE;
        break;
        
    case DONE:          // 完成
        return Z_STREAM_END;
}
```

## 五、刷新机制详解

### 5.1 Flush参数

```c
#define Z_NO_FLUSH      0       /* 不刷新 */
#define Z_PARTIAL_FLUSH 1      /* 部分刷新 */
#define Z_SYNC_FLUSH    2       /* 同步刷新 */
#define Z_FULL_FLUSH    3       /* 完全刷新 */
#define Z_FINISH        4       /* 完成压缩 */
#define Z_BLOCK         5       /* 块刷新 */
#define Z_TREES         6       /* 树刷新 */
```

### 5.2 各Flush模式对比

#### Z_NO_FLUSH

```c
// 默认模式，继续压缩
strm.avail_in = 1024;
strm.next_in = input_data;
ret = deflate(&strm, Z_NO_FLUSH);
// 数据继续积累到内部缓冲区
// 不会输出部分结果
```

**特点**:
- 不强制输出
- 继续积累数据
- 最大化压缩率

#### Z_PARTIAL_FLUSH

```c
// 部分刷新，输出当前块
ret = deflate(&strm, Z_PARTIAL_FLUSH);
// 输出当前压缩块
// 不清除哈希表
// 可以继续压缩
```

**特点**:
- 输出当前块
- 保持压缩状态
- 适合流式处理

#### Z_SYNC_FLUSH

```c
// 同步刷新，对齐字节边界
ret = deflate(&strm, Z_SYNC_FLUSH);
// 输出当前块并填充到字节边界
// 确保解压器可以同步
```

**特点**:
- 输出当前块
- 对齐到字节边界
- 允许随机访问

#### Z_FULL_FLUSH

```c
// 完全刷新，清除所有状态
ret = deflate(&strm, Z_FULL_FLUSH);
// 输出当前块
// 清除哈希表
// 清空滑动窗口
```

**特点**:
- 重置压缩状态
- 清空滑动窗口
- 适合错误恢复

#### Z_FINISH

```c
// 完成压缩
do {
    ret = deflate(&strm, Z_FINISH);
    // 处理输出...
} while (ret == Z_OK);
// ret == Z_STREAM_END 表示完成
```

**特点**:
- 最后一个flush参数
- 输出所有待定数据
- 关闭流

### 5.3 Flush策略选择

```c
// 场景1: 文件压缩
deflate(&strm, Z_NO_FLUSH);    // 压缩文件内容
...
deflate(&strm, Z_FINISH);      // 完成压缩

// 场景2: 网络实时传输
deflate(&strm, Z_NO_FLUSH);    // 积累数据
if (need_send()) {
    deflate(&strm, Z_SYNC_FLUSH);  // 发送数据
}

// 场景3: 随机访问
deflate(&strm, Z_FULL_FLUSH);  // 创建检查点
// 可以从检查点恢复

// 场景4: 错误恢复
if (error_detected) {
    deflate(&strm, Z_FULL_FLUSH);  // 重置状态
}
```

## 六、错误处理机制

### 6.1 返回码处理

```c
int ret = deflate(&strm, flush);

switch (ret) {
    case Z_OK:
        // 成功，继续处理
        break;
        
    case Z_STREAM_END:
        // 流结束
        // 对于Z_FINISH，这是正常结果
        break;
        
    case Z_STREAM_ERROR:
        // 流状态错误
        // 例如：next_in或next_out为NULL
        printf("Stream error: %s\n", strm.msg);
        break;
        
    case Z_DATA_ERROR:
        // 数据错误
        // 例如：校验和失败
        printf("Data error: %s\n", strm.msg);
        break;
        
    case Z_MEM_ERROR:
        // 内存不足
        printf("Memory error: %s\n", strm.msg);
        break;
        
    case Z_BUF_ERROR:
        // 缓冲区错误
        // 输入或输出缓冲区太小
        printf("Buffer error\n");
        break;
}
```

### 6.2 错误恢复

```c
int compress_with_recovery(z_stream *strm) {
    int ret;
    int retry_count = 0;
    const int max_retries = 3;
    
    while (retry_count < max_retries) {
        ret = deflate(strm, Z_NO_FLUSH);
        
        if (ret == Z_OK) {
            return Z_OK;
        }
        
        if (ret == Z_STREAM_ERROR) {
            // 尝试恢复
            deflateReset(strm);
            retry_count++;
            continue;
        }
        
        if (ret == Z_BUF_ERROR) {
            // 增大缓冲区
            if (resize_buffers()) {
                continue;
            }
        }
        
        // 其他错误无法恢复
        return ret;
    }
    
    return ret;
}
```

### 7.3 管道优化

```c
// 使用管道连接多个处理阶段
void compress_pipeline() {
    int pipe1[2], pipe2[2];
    pid_t pid1, pid2;
    
    pipe(pipe1);  // 输入 -> 压缩
    pipe(pipe2);  // 压缩 -> 输出
    
    // 进程1: 压缩
    pid1 = fork();
    if (pid1 == 0) {
        close(pipe1[1]);   // 关闭写入端
        close(pipe2[0]);   // 关闭读取端
        
        // 从pipe1读取，压缩后写入pipe2
        compress_stream_from_fd(pipe1[0], pipe2[1]);
        
        exit(0);
    }
    
    // 进程2: 传输
    pid2 = fork();
    if (pid2 == 0) {
        close(pipe1[0]);
        close(pipe1[1]);
        close(pipe2[1]);
        
        // 从pipe2读取，传输到网络
        transmit_from_fd(pipe2[0]);
        
        exit(0);
    }
    
    // 主进程: 写入输入数据
    close(pipe1[0]);
    close(pipe2[0]);
    close(pipe2[1]);
    
    // 写入数据到pipe1
    write_to_pipe(pipe1[1]);
}
```

## 八、实际应用案例

### 8.1 HTTP流式压缩

```c
// Web服务器压缩响应
void compress_http_response(FILE *socket, 
                            const char *content,
                            size_t content_len) {
    z_stream strm;
    unsigned char out[CHUNK_SIZE];
    int ret;
    
    // 初始化
    deflateInit2(&strm, Z_DEFAULT_COMPRESSION,
                 Z_DEFLATED, -15, 8, Z_DEFAULT_STRATEGY);
    
    // 设置gzip头部
    const char gzip_header[] = "\x1f\x8b\x08\x00\x00\x00\x00\x00\x00\xff";
    write(socket, gzip_header, sizeof(gzip_header) - 1);
    
    // 压缩数据
    strm.next_in = (unsigned char *)content;
    strm.avail_in = content_len;
    
    do {
        strm.avail_out = CHUNK_SIZE;
        strm.next_out = out;
        ret = deflate(&strm, Z_FINISH);
        
        if (ret == Z_STREAM_ERROR) {
            break;
        }
        
        size_t have = CHUNK_SIZE - strm.avail_out;
        write(socket, out, have);
    } while (strm.avail_out == 0);
    
    // 写入gzip尾部
    unsigned char trailer[8];
    // ... 构造trailer ...
    write(socket, trailer, 8);
    
    deflateEnd(&strm);
}
```

```c
    lc->compressed_file = fopen(compressed_path, "wb");
    
    // 初始化deflate
    lc->strm.zalloc = Z_NULL;
    lc->strm.zfree = Z_NULL;
    lc->strm.opaque = Z_NULL;
    deflateInit(&lc->strm, Z_DEFAULT_COMPRESSION);
    
    lc->initialized = 1;
    return lc;
}

void write_log(LogCompressor *lc, const char *message) {
    if (!lc->initialized) return;
    
    // 写入临时文件
    fprintf(lc->log_file, "%s\n", message);
    fflush(lc->log_file);
    
    // 定期压缩
    if (ftell(lc->log_file) > CHUNK_SIZE / 2) {
        compress_log_buffer(lc);
    }
}

void compress_log_buffer(LogCompressor *lc) {
    rewind(lc->log_file);
    
    size_t read_size;
    do {
        lc->strm.next_out = lc->buffer;
        lc->strm.avail_out = CHUNK_SIZE;
        
        read_size = fread(lc->buffer, 1, CHUNK_SIZE, lc->log_file);
        lc->strm.next_in = lc->buffer;
        lc->strm.avail_in = read_size;
        
        int ret = deflate(&lc->strm, Z_SYNC_FLUSH);
        if (ret == Z_OK) {
            size_t have = CHUNK_SIZE - lc->strm.avail_out;
            fwrite(lc->buffer, 1, have, lc->compressed_file);
        }
    } while (read_size > 0);
    
    ftruncate(fileno(lc->log_file), 0);
    rewind(lc->log_file);
}
```

### 9.1 内存占用分析

```c
// 流式压缩内存占用
Memory = sizeof(z_stream)      // 约100字节
         + window_size        // 32KB - 256KB

         + output_buffer      // 用户指定
         + input_buffer       // 用户指定
```

### 9.3 优化建议

1. **缓冲区大小**: 16KB-64KB是最佳范围
2. **批量处理**: 不要每次只处理几字节
3. **异步I/O**: 使用非阻塞I/O提升吞吐
4. **双缓冲**: 边读边写，减少等待时间

## 十、总结

流式压缩的核心优势：

1. **内存高效**: 恒定内存占用，处理大文件
2. **实时处理**: 边读边压缩，低延迟
3. **灵活性**: 支持暂停、恢复、管道操作

关键要点：

1. 正确管理avail_in/avail_out
2. 选择合适的flush参数
3. 完善的错误处理
4. 优化缓冲区大小

适用场景：

- 大文件压缩/解压
- 网络数据传输
- 实时数据流处理
- 日志压缩
