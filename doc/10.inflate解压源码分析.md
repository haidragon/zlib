# inflate解压源码分析

## 概述

inflate是zlib库中的解压缩函数，用于解压使用DEFLATE算法压缩的数据。本文档详细分析inflate的源码实现。

## 核心函数分析

### inflateInit_

```c
int inflateInit2_(z_streamp strm, int windowBits, const char *version, int stream_size)
{
    struct inflate_state FAR *state;

    if (version == Z_NULL || version[0] != ZLIB_VERSION[0] ||
        stream_size != sizeof(z_stream))
        return Z_VERSION_ERROR;

    if (strm == Z_NULL)
        return Z_STREAM_ERROR;

    strm->msg = Z_NULL;

    if (strm->zalloc == (alloc_func)0) {
        strm->zalloc = zcalloc;
        strm->opaque = (voidpf)0;
    }
    if (strm->zfree == (free_func)0)
        strm->zfree = zcfree;

    state = (struct inflate_state FAR *)
            ZALLOC(strm, 1, sizeof(struct inflate_state));
    if (state == Z_NULL)
        return Z_MEM_ERROR;

    Tracev((stderr, "inflate: allocated\n"));
    strm->state = (struct internal_state FAR *)state;
    state->window = Z_NULL;
    state->wbits = windowBits;
    state->wsize = 1U << windowBits;
    state->whave = 0;
    state->wnext = 0;

    return inflateReset(strm);
}
```

**功能说明：**
- 初始化inflate状态
- 分配内存
- 设置窗口大小 (由windowBits决定)
- 调用inflateReset重置状态

### inflateReset

```c
int inflateReset(z_streamp strm)
{
    struct inflate_state FAR *state;

    if (strm == Z_NULL || strm->state == Z_NULL)
        return Z_STREAM_ERROR;

    state = (struct inflate_state FAR *)strm->state;
    strm->total_in = strm->total_out = 0;
    strm->msg = Z_NULL;

    state->mode = HEAD;
    state->last = 0;
    state->wrap = 1;  // 默认zlib格式
    state->havedict = 0;
    state->flags = 0;
    state->dmax = 32768U;
    state->head = Z_NULL;
    state->hold = 0;
    state->bits = 0;
    state->lencode = state->distcode = state->next = state->codes;

    Tracev((stderr, "inflate: reset\n"));
    return Z_OK;
}
```

**功能说明：**
- 重置inflate状态到初始值
- 清空输入/输出计数器
- 设置初始模式为HEAD

### inflate - 主函数

```c
int inflate(z_streamp strm, int flush)
{
    struct inflate_state FAR *state;
    unsigned char FAR *next;
    unsigned char FAR *put;
    unsigned have, left;
    unsigned long hold;
    unsigned bits;
    code const FAR *lcode;
    code const FAR *dcode;
    unsigned lmask;
    unsigned dmask;
    code here;
    unsigned len;
    unsigned dist;
    static const unsigned short order[19] =
        {16, 17, 18, 0, 8, 7, 9, 6, 10, 5, 11, 4, 12, 3, 13, 2, 14, 1, 15};

    if (strm == Z_NULL || strm->state == Z_NULL || strm->next_out == Z_NULL ||
        (strm->next_in == Z_NULL && strm->avail_in != 0))
        return Z_STREAM_ERROR;

    state = (struct inflate_state FAR *)strm->state;

    if (state->mode == TYPE)
        state->mode = TYPEDO;

    // 主循环
    for (;;) {
        switch (state->mode) {
        case HEAD:
            // 处理头部
            break;

        case FLAGS:
            // 处理标志
            break;

        case DICTID:
            // 处理字典ID
            break;

        case DICT:
            // 处理字典
            break;

        case TYPE:
            // 读取块类型
            break;

        case TYPEDO:
            // 处理存储块
            break;

        case STORED:
            // 处理存储块数据
            break;

        case COPY:
            // 复制数据
            break;

        case TABLE:
            // 读取Huffman表
            break;

        case LENLENS:
            // 读取长度编码长度
            break;

        case CODELENS:
            // 读取编码长度
            break;

        case LEN_:
            // 处理长度
            break;

        case LENEXT:
            // 长度扩展
            break;

        case DIST:
            // 处理距离
            break;

        case DISTEXT:
            // 距离扩展
            break;

        case MATCH:
            // 匹配复制
            break;

        case LIT:
            // 输出字面量
            break;

        case CHECK:
            // 校验和检查
            break;

        case DONE:
            // 完成
            return Z_STREAM_END;

        case BAD:
            // 错误
            return Z_DATA_ERROR;

        case MEM:
            // 内存错误
            return Z_MEM_ERROR;

        case SYNC:
            // 同步
            break;

        default:
            return Z_STREAM_ERROR;
        }
    }
}
```

## 关键状态处理

### 1. HEAD状态 - 头部处理

```c
case HEAD:
    if (state->wrap == 0) {
        state->mode = TYPEDO;
        break;
    }

    // 读取2字节头部
    NEEDBITS(16);

    // 检查zlib魔数
    if ((BITS(8) << 8) + (hold >> 8) % 31 != 0) {
        strm->msg = (char *)"incorrect header check";
        state->mode = BAD;
        break;
    }

    // 检查压缩方法
    if (BITS(4) != Z_DEFLATED) {
        strm->msg = (char *)"unknown compression method";
        state->mode = BAD;
        break;
    }

    // 获取窗口大小
    state->wbits = ((BITS(4) + 8) << 4) + 8;
    hold >>= 8;
    bits -= 8;

    state->mode = FLAGS;
    break;
```

### 2. TYPE状态 - 读取块类型

```c
case TYPE:
    // 读取3位块类型
    NEEDBITS(3);
    state->last = BITS(1);
    DROPBITS(1);
    switch (BITS(2)) {
    case 0:                             // 存储块
        Tracev((stderr, "inflate:     stored block%s\n",
                state->last ? " (last)" : ""));
        state->mode = STORED;
        break;
    case 1:                             // 固定Huffman编码
        fixedtables(state);
        Tracev((stderr, "inflate:     fixed codes block%s\n",
                state->last ? " (last)" : ""));
        state->mode = LEN_;             // 立即开始解码
        break;
    case 2:                             // 动态Huffman编码
        Tracev((stderr, "inflate:     dynamic codes block%s\n",
                state->last ? " (last)" : ""));
        state->mode = TABLE;
        break;
    case 3:                             // 保留
        strm->msg = (char *)"invalid block type";
        state->mode = BAD;
        break;
    }
    DROPBITS(2);
    break;
```

### 3. TABLE状态 - 读取动态Huffman表

```c
case TABLE:
    // 读取HLIT, HDIST, HCLEN
    NEEDBITS(14);
    state->nlen = BITS(5) + 257;
    DROPBITS(5);
    state->ndist = BITS(5) + 1;
    DROPBITS(5);
    state->ncode = BITS(4) + 4;
    DROPBITS(4);

    // 读取编码长度表
    state->mode = LENLENS;
    break;

case LENLENS:
    // 读取ncode个编码长度
    while (state->have < state->ncode) {
        NEEDBITS(3);
        state->lens[order[state->have++]] = BITS(3);
        DROPBITS(3);
    }
    state->have = 0;

    // 处理长度编码
    state->mode = CODELENS;
    break;

case CODELENS:
    // 解码长度和距离编码表
    while (state->have < state->nlen + state->ndist) {
        for (;;) {
            here = state->len;
            if ((unsigned)(here.bits) <= bits) break;
            PULLBYTE();
        }

        if (here.val < 16) {
            state->lens[state->have++] = here.val;
        } else {
            // 处理重复值
            if (here.val == 16) {
                NEEDBITS(2);
                len = BITS(2) + 3;
                DROPBITS(2);
            } else if (here.val == 17) {
                NEEDBITS(3);
                len = BITS(3) + 3;
                DROPBITS(3);
            } else {
                NEEDBITS(7);
                len = BITS(7) + 11;
                DROPBITS(7);
            }

            // 填充重复值
            for (; len > 0; len--) {
                state->lens[state->have++] = 0;
            }
        }
    }

    // 构建Huffman表
    state->next = state->codes;
    state->lencode = next;
    len = state->nlen;
    ret = inflate_table(LENS, state->lens, len,
                        &(state->next), &(state->lenbits), state->work);

    if (ret) {
        strm->msg = (char *)"invalid literal/lengths set";
        state->mode = BAD;
        break;
    }

    state->distcode = next;
    len = state->ndist;
    ret = inflate_table(DISTS, state + lens + state->nlen, len,
                        &(state->next), &(state->distbits), state->work);

    if (ret) {
        strm->msg = (char *)"invalid distances set";
        state->mode = BAD;
        break;
    }

    Tracev((stderr, "inflate:       codes ok\n"));
    state->mode = LEN_;
    break;
```

### 4. LEN_状态 - 解码长度/字面量

```c
case LEN_:
    // 解码Huffman编码
    for (;;) {
        here = state->lencode[BITS(state->lenbits)];
        if ((unsigned)(here.bits) <= bits) break;
        PULLBYTE();
    }

    if (here.op && (here.op & 0xf0) == 0) {
        last = here;
        for (;;) {
            here = state->lencode[last.val +
                    (BITS(last.bits + last.op) >> last.bits)];
            if ((unsigned)(last.bits + here.bits) <= bits) break;
            PULLBYTE();
        }
        DROPBITS(last.bits);
        state->hold = hold;
        state->bits = bits;
    }

    DROPBITS(here.bits);

    state->length = here.val;

    if ((int)(here.op) == 0) {
        Tracevv((stderr, here.val >= 0x20 && here.val < 0x7f ?
                "inflate:         literal '%c'\n" :
                "inflate:         literal 0x%02x\n", here.val));
        state->mode = LIT;
        break;
    }

    if (here.op & 32) {
        Tracevv((stderr, "inflate:         end of block\n"));
        state->mode = TYPE;
        break;
    }

    if (here.op & 64) {
        strm->msg = (char *)"invalid literal/length code";
        state->mode = BAD;
        break;
    }

    state->extra = (unsigned)(here.op) & 15;
    state->mode = LENEXT;
    break;

case LENEXT:
    if (state->extra) {
        NEEDBITS(state->extra);
        state->length += BITS(state->extra);
        DROPBITS(state->extra);
    }
    Tracevv((stderr, "inflate:         length %u\n", state->length));
    state->mode = DIST;
    break;
```

### 5. DIST状态 - 解码距离

```c
case DIST:
    for (;;) {
        here = state->distcode[BITS(state->distbits)];
        if ((unsigned)(here.bits) <= bits) break;
        PULLBYTE();
    }

    if ((here.op & 0xf0) == 0) {
        last = here;
        for (;;) {
            here = state->distcode[last.val +
                    (BITS(last.bits + last.op) >> last.bits)];
            if ((unsigned)(last.bits + here.bits) <= bits) break;
            PULLBYTE();
        }
        DROPBITS(last.bits);
        state->hold = hold;
        state->bits = bits;
    }

    DROPBITS(here.bits);

    if (here.op & 64) {
        strm->msg = (char *)"invalid distance code";
        state->mode = BAD;
        break;
    }

    state->offset = here.val;
    state->extra = (unsigned)(here.op) & 15;
    state->mode = DISTEXT;
    break;

case DISTEXT:
    if (state->extra) {
        NEEDBITS(state->extra);
        state->offset += BITS(state->extra);
        DROPBITS(state->extra);
    }

    // 检查距离是否在窗口内
    if (state->offset > state->whave + state->wnext - state->whave) {
        strm->msg = (char *)"invalid distance too far back";
        state->mode = BAD;
        break;
    }

    Tracevv((stderr, "inflate:         distance %u\n", state->offset));
    state->mode = MATCH;
    break;
```

### 6. MATCH状态 - 复制匹配数据

```c
case MATCH:
    // 计算复制源位置
    copy = state->offset;
    if (copy > state->wnext) {
        copy -= state->wnext;
        from = state->window + state->wsize - copy;
    } else {
        from = state->window + state->wnext - copy;
    }

    // 复制数据
    if (copy - left > 0) {
        // 需要环绕
        do {
            *put++ = *from++;
            left--;
        } while (left && from < state->window + state->wsize);
    }

    // 直接复制
    while (left > 0) {
        *put++ = *from++;
        left--;
    }

    state->mode = LEN_;
    break;
```

### 7. LIT状态 - 输出字面量

```c
case LIT:
    // 输出字面量
    *put++ = (unsigned char)(state->length);
    state->have--;
    state->mode = LEN_;
    break;
```

### 8. CHECK状态 - 检查校验和

```c
case CHECK:
    if (state->wrap) {
        NEEDBITS(32);
        state->check = ZSWAP32(hold);
        strm->adler = state->check;

        // 验证校验和
        if (state->flags) {
            // GZIP CRC32
            if (state->check != state->total) {
                strm->msg = (char *)"incorrect data check";
                state->mode = BAD;
                break;
            }
        } else {
            // zlib Adler32
            if (state->check != state->total) {
                strm->msg = (char *)"incorrect data check";
                state->mode = BAD;
                break;
            }
        }

        Tracev((stderr, "inflate:   check matches trailer\n"));
    }

    state->mode = DONE;
    break;
```

## 辅助函数

### inflate_table - 构建Huffman表

```c
int inflate_table(codetype type, unsigned short FAR *lens,
                  unsigned codes, code FAR * FAR *table,
                  unsigned FAR *bits, unsigned short FAR *work)
{
    unsigned len;           // 当前代码长度
    unsigned sym;           // 符号索引
    unsigned min, max;       // 最小/最大代码长度
    unsigned root;          // 根表大小
    unsigned curr;          // 当前表大小
    unsigned drop;          // 丢弃的索引
    int left;               // 剩余的代码
    unsigned used;         // 已使用的表项
    unsigned huff;          // Huffman代码
    unsigned incr;          // 增量
    unsigned fill;          // 填充表项
    unsigned low;           // 低位
    unsigned mask;          // 掩码
    code *next;             // 下一个表项
    const unsigned short FAR *base;     // 基础值表
    const unsigned short FAR *extra;    // 额外值表

    // 步骤1: 计算代码长度分布
    for (len = 0; len <= MAXBITS; len++)
        count[len] = 0;
    for (sym = 0; sym < codes; sym++)
        count[lens[sym]]++;

    // 步骤2: 计算每个长度的代码范围
    root = bits[0];
    for (len = 1; len <= MAXBITS; len++) {
        root += count[len] << len;
        if (root > (1U << len)) return 1;  // 超出范围
    }
    if (root > (1U << MAXBITS)) return 1;

    // 步骤3: 分配Huffman代码
    next = *table;
    left = root;
    for (len = 1; len <= MAXBITS; len++) {
        left -= count[len];
        if (left < 0) return -1;
        offs[len] = left;
    }

    for (sym = 0; sym < codes; sym++) {
        if (lens[sym] == 0) continue;

        huff = offs[lens[sym]]++;
        huff >>= MAXBITS - lens[sym];

        // 填充表
        if (lens[sym] <= root) {
            next[huff].op = (unsigned char)len;
            next[huff].bits = (unsigned char)lens[sym];
            next[huff].val = (unsigned short)sym;
        } else {
            // 需要子表
        }
    }

    // 步骤4: 优化表大小
    *bits = root;

    return 0;
}
```

### fixedtables - 固定Huffman表

```c
void fixedtables(struct inflate_state FAR *state)
{
    static code lenfix[512];
    static code distfix[32];
    static code *lenfix_initialized = NULL;
    static code *distfix_initialized = NULL;

    // 初始化长度编码表 (只执行一次)
    if (lenfix_initialized != lenfix) {
        unsigned sym, bits;
        static const unsigned short lencnt[MAXBITS+1], lensym[MAXBITS+1];
        static const unsigned short distcnt[MAXBITS+1], distsym[MAXBITS+1];

        // 构建长度编码表
        sym = 0;
        while (sym < 144) state->lens[sym++] = 8;
        while (sym < 256) state->lens[sym++] = 9;
        while (sym < 280) state->lens[sym++] = 7;
        while (sym < 288) state->lens[sym++] = 8;

        inflate_table(LENS, state->lens, 288, &(state->next),
                      &(state->lenbits), state->work);

        lenfix_initialized = lenfix;
    }

    state->lencode = lenfix;

    // 初始化距离编码表
    if (distfix_initialized != distfix) {
        unsigned sym;
        sym = 0;
        while (sym < 32) state->lens[sym++] = 5;

        inflate_table(DISTS, state->lens, 32, &(state->next),
                      &(state->distbits), state->work);

        distfix_initialized = distfix;
    }

    state->distcode = distfix;
}
```

## 宏定义

```c
// 读取位
#define BITS(n) \
    ((hold) & ((1U << (n)) - 1))

// 丢弃位
#define DROPBITS(n) \
    do { \
        hold >>= (n); \
        bits -= (unsigned)(n); \
    } while (0)

// 需要更多位
#define NEEDBITS(n) \
    do { \
        while (bits < (unsigned)(n)) \
            PULLBYTE(); \
    } while (0)

// 拉取一个字节
#define PULLBYTE() \
    do { \
        if (have == 0) goto inf_leave; \
        have--; \
        hold += (unsigned long)(*next++) << bits; \
        bits += 8; \
    } while (0)

// 复制字节
#define ROOM() \
    do { \
        if (left == 0) goto inf_leave; \
        left--; \
    } while (0)
```

## 完整使用示例

```c
#include <zlib.h>
#include <stdio.h>
#include <stdlib.h>

int decompress_data(const unsigned char *compressed,
                    size_t compressed_len,
                    unsigned char **out_data,
                    size_t *out_len) {
    int ret;
    z_stream stream;

    // 初始化
    stream.zalloc = Z_NULL;
    stream.zfree = Z_NULL;
    stream.opaque = Z_NULL;
    stream.avail_in = compressed_len;
    stream.next_in = (Bytef *)compressed;

    ret = inflateInit(&stream);
    if (ret != Z_OK) {
        return ret;
    }

    // 分配输出缓冲区
    size_t out_capacity = 4096;
    *out_data = (unsigned char *)malloc(out_capacity);
    if (!*out_data) {
        inflateEnd(&stream);
        return Z_MEM_ERROR;
    }

    stream.avail_out = out_capacity;
    stream.next_out = *out_data;

    // 解压缩
    ret = inflate(&stream, Z_FINISH);
    if (ret != Z_STREAM_END) {
        free(*out_data);
        inflateEnd(&stream);
        return ret;
    }

    *out_len = out_capacity - stream.avail_out;

    // 清理
    inflateEnd(&stream);

    return Z_OK;
}

int main(void) {
    const char *test_data = "Hello, World! This is a test string.";
    unsigned char compressed[1024];
    uLongf compressed_len = sizeof(compressed);

    // 先压缩
    compress(compressed, &compressed_len,
             (Bytef *)test_data, strlen(test_data) + 1);

    // 再解压缩
    unsigned char *decompressed = NULL;
    size_t decompressed_len = 0;

    int ret = decompress_data(compressed, compressed_len,
                              &decompressed, &decompressed_len);
    if (ret == Z_OK) {
        printf("解压缩成功: %s\n", (char *)decompressed);
        free(decompressed);
    } else {
        printf("解压缩失败: %d\n", ret);
    }

    return 0;
}
```

### 3. 流式解压缩

```c
// 分块处理大文件
#define CHUNK_SIZE 16384

int decompress_stream(FILE *in, FILE *out) {
    z_stream stream;
    unsigned char in_buf[CHUNK_SIZE];
    unsigned char out_buf[CHUNK_SIZE];
    int ret;

    stream.zalloc = Z_NULL;
    stream.zfree = Z_NULL;
    stream.opaque = Z_NULL;

    ret = inflateInit(&stream);
    if (ret != Z_OK) return ret;

    do {
        stream.avail_in = fread(in_buf, 1, CHUNK_SIZE, in);
        if (ferror(in)) {
            inflateEnd(&stream);
            return Z_ERRNO;
        }

        if (stream.avail_in == 0) break;

        stream.next_in = in_buf;

        do {
            stream.avail_out = CHUNK_SIZE;
            stream.next_out = out_buf;

            ret = inflate(&stream, Z_NO_FLUSH);

            if (ret == Z_STREAM_ERROR) {
                inflateEnd(&stream);
                return ret;
            }

            unsigned have = CHUNK_SIZE - stream.avail_out;
            if (fwrite(out_buf, 1, have, out) != have || ferror(out)) {
                inflateEnd(&stream);
                return Z_ERRNO;
            }
        } while (stream.avail_out == 0);
    } while (ret != Z_STREAM_END);

    inflateEnd(&stream);
    return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
}
```

## 总结

inflate解压缩的核心流程：

1. **初始化**: inflateInit2 设置窗口大小和包装格式
2. **头部处理**: 解析zlib/gzip头部
3. **块类型判断**: 读取BFINAL和BTYPE
4. **Huffman解码**:
   - 固定编码: 使用预定义表
   - 动态编码: 解码Huffman表
5. **数据解压**: 
   - 字面量: 直接输出
   - 匹配: 从滑动窗口复制
6. **校验和验证**: 检查数据完整性
7. **清理**: inflateEnd释放资源

关键点：
- 使用状态机处理不同阶段
- 滑动窗口实现LZ77匹配
- Huffman编码高效解码
- 支持流式处理
- 完善的错误处理
